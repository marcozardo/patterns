
model *New_Model()

  // Compartments and Species:
  compartment compartment_;
  species P in compartment_, D in compartment_, I in compartment_, $S in compartment_;

  // Assignment Rules:
  S := P + I;
  f_D := 1 + tanh((D - delta)/omega);

  // Reactions:
  reaction_1:  => P; compartment_*alpha*P*(1 - P/k_p);
  reaction_2: P => ; compartment_*beta*D*(P/(P + 0.01));
  reaction_3:  => D; compartment_*gamma_*P;
  reaction_4: D => ; compartment_*theta*D;
  reaction_5:  => I; compartment_*epsilon*f_D;
  reaction_6: I => ; compartment_*rho*I;

  // Species initializations:
  P = 0.01/compartment_;
  D = 0;
  I = 0;

  // Compartment initializations:
  compartment_ = 1;

  // Variable initializations:
  delta = 4.27;
  omega = 0.13;
  alpha = 3.63;
  k_p = 3.23;
  beta = 1;
  gamma_ = 0.51;
  theta = 0.01;
  epsilon = 6.81;
  rho = 1.82;

  // Other declarations:
  var f_D;
  const compartment_, delta, omega, alpha, k_p, beta, gamma_, theta, epsilon;
  const rho;

  // Unit definitions:
  unit volume = 1e-3 litre;
  unit time_unit = 8.64e4 second;
  unit substance = 1e-3 mole;

  // Display Names:
  time_unit is "time";
  compartment_ is "Mice";
  P is "Virus Pathogenicity (P)";
  D is "Antiviral Immune defense (D)";
  I is "Inflammation (I)";
  S is "Clinical Score (S)";
  delta is "delta'";
  omega is "omega'";
  gamma_ is "gamma'";
  reaction_1 is "Infection of Virus";
  reaction_2 is "Early immune response to the virus";
  reaction_3 is "Activation of the defense system";
  reaction_4 is "Decay in immunity";
  reaction_5 is "Inflammatory response by the host";
  reaction_6 is "Anti-inflammatory response by the host";
  
end
