model *BIOMD0000000139___Model_of_Calcium_handling_and_contraction_in_uterine_smooth_muscle_cell()
  // Compartments and Species:
  compartment cyt, SR;
  species Ca_cyt in cyt, Ca_SR in SR, IP3 in cyt, R in cyt, IP3R in cyt, CaR in cyt, CaIP3R in cyt, Ca2R in cyt, Ca2IP3R in cyt;
  species h in cyt, PLC in cyt, G in cyt, DAG in cyt, PKC in cyt, MLCK in cyt, MLCK_a in cyt, MLC in cyt, MLC_p in cyt;
  species Force in cyt;

  // Assignment Rules:
  J_IP3R := (v1*(Ca2IP3R*(Ca_SR - Ca_cyt) + v2*(CaIP3R*(Ca_SR - Ca_cyt)) + v3*(IP3R*(Ca_SR - Ca_cyt))));
  J_leak := v4*(Ca_SR - Ca_cyt);
  J_out := v5;
  J_pump := v6*Ca_cyt^2/(Ca_cyt^2 + k_pump^2);
  k_h := k_minus + (k_plus - k_minus)*Ca_cyt^4/(Ca_cyt^4 + K_a^4);
  k_1 := k1*IP3;
  k_2 := k2*Ca_cyt;
  k_3 := k3*IP3;
  k_4 := k4*Ca_cyt;
  k_minus1 := k_minus1;
  k_minus2 := k_minus2;
  k_minus3 := k_minus3;
  k_minus4 := k_minus4;
  PLC_star := PLC*G;
  J_PLC := v7*PLC_star;
  J_degr := v8*IP3;
  k_PKC := k9*DAG;
  k_PKC_minus := k10;
  k_MLCK := k11*Ca_cyt^2/(Ca_cyt^2 + K_MLCK^2);
  k_MLCK_minus := k12;
  k_MLC := k13*MLCK_a;
  k_MLC_minus := k14;
  Force := F0 + (F_max - F0)*MLC_p/(MLC_p + K_force);

  // Reactions:
  R_binding_IP3: R + IP3 -> IP3R; k_1*R*IP3 - k_minus1*IP3R;
  R_binding_Ca: R + Ca_cyt -> CaR; k_2*R*Ca_cyt - k_minus2*CaR;
  IP3R_binding_Ca: IP3R + Ca_cyt -> CaIP3R; k_4*IP3R*Ca_cyt - k_minus4*CaIP3R;
  CaR_binding_IP3: CaR + IP3 -> CaIP3R; k_3*CaR*IP3 - k_minus3*CaIP3R;
  CaIP3R_binding_Ca: CaIP3R + Ca_cyt -> Ca2IP3R; k_2*CaIP3R*Ca_cyt - k_minus2*Ca2IP3R;
  CaR_binding_Ca: CaR + Ca_cyt -> Ca2R; k_4*CaR*Ca_cyt - k_minus4*Ca2R;
  Ca2R_binding_IP3: Ca2R + IP3 -> Ca2IP3R; k_1*Ca2R*IP3 - k_minus1*Ca2IP3R;
  h_gate: h -> ; k_h*h;
  IP3_production:  => IP3; J_PLC;
  IP3_degradation: IP3 => ; J_degr;
  G_activation: G -> ; k_G_act*G;
  G_deactivation:  => G; k_G_inact*(G_tot - G);
  PLC_activation: PLC -> ; k_PLC_act*PLC;
  PLC_deactivation:  => PLC; k_PLC_inact*(PLC_tot - PLC);
  DAG_production:  => DAG; k_DAG*PLC_star;
  DAG_degradation: DAG => ; k_DAG_minus*DAG;
  PKC_activation: PKC -> ; k_PKC*PKC;
  PKC_deactivation:  => PKC; k_PKC_minus*(PKC_tot - PKC);
  MLCK_activation: MLCK -> MLCK_a; k_MLCK*MLCK - k_MLCK_minus*MLCK_a;
  MLC_phosphorylation: MLC -> MLC_p; k_MLC*MLC - k_MLC_minus*MLC_p;
  Ca_cyt_flux:  => Ca_cyt; (J_IP3R + J_leak - J_pump + J_out)/cyt;
  Ca_SR_flux:  => Ca_SR; (-J_IP3R - J_leak + J_pump)/SR;

  // Species initializations:
  Ca_cyt = 0.1;
  Ca_SR = 10;

  v1 = 20;

  v3 = 0.04;
  v4 = 0.002;
  v5 = 0.02;
  v6 = 0.4;
  k_pump = 0.2;

  // Other declarations:
  var J_IP3R, J_leak, J_out, J_pump, k_h, k_1, k_2, k_3, k_4, k_minus1, k_minus2;
  var k_minus3, k_minus4, PLC_star, J_PLC, J_degr, k_PKC, k_PKC_minus, k_MLCK;
  var k_MLCK_minus, k_MLC, k_MLC_minus, Force;
  const cyt, SR, v1, v2, v3, v4, v5, v6, k_pump, k_minus, k_plus, K_a, k1, k2;
  const k3, k4, k_minus1, k_minus2, k_minus3, k_minus4, v7, v8, k_G_act, k_G_inact;
  const G_tot, k_PLC_act, k_PLC_inact, PLC_tot, k_DAG, k_DAG_minus, k9, k10;
  const PKC_tot, k11, k12, K_MLCK, k13, k14, F0, F_max, K_force;
