  species G in compartment_, PLC in compartment_, PLC_star in compartment_;
  species PLC_star_G in compartment_, R in compartment_, R_star in compartment_;
  species R_star_G in compartment_, G_star in compartment_;

  // Assignment Rules:
  Jbeta := vbeta * (IP3^2 / (Kbeta^2 + IP3^2));
  Jdelta := vdelta * (Ca^2 / (Kdelta^2 + Ca^2));
  J3k := V3k * (Ca^4 / (K3k^4 + Ca^4)) * (IP3 / (Kd + IP3));
  J5p := V5p * (IP3 / (K5p + IP3));
  PLC_tot := PLC + PLC_star + PLC_star_G;
  R_tot := R + R_star + R_star_G;
  G_tot := G + G_star + PLC_star_G + R_star_G;

  // Reactions:
  PIP2_Hydrolysis: PIP2 => IP3; compartment_*(k1*PLC_star*PIP2);
  IP3_Degradation: IP3 => ; compartment_*(k2*IP3);
  Ca_Release:  => Ca; compartment_*(k3*IP3*Ca);
  Ca_Extrusion: Ca => ; compartment_*(k4*Ca);
  G_Activation_by_PLC_star: G => G_star; compartment_*(k5*PLC_star*G - k6*G_star);
  G_Activation_by_R_star: G => G_star; compartment_*(k7*R_star*G - k8*G_star);
  PLC_Activation: PLC => PLC_star; compartment_*(k9*Ca*PLC - k10*PLC_star);
  PLC_star_G_Complex_Formation: PLC_star + G => PLC_star_G; compartment_*(k11*PLC_star*G - k12*PLC_star_G);
  R_Activation: R => R_star; compartment_*(k13*Ca*R - k14*R_star);
  R_star_G_Complex_Formation: R_star + G => R_star_G; compartment_*(k15*R_star*G - k16*R_star_G);
  IP3_Synthesis:  => IP3; compartment_*Jbeta;
  IP3_Dephosphorylation: IP3 => ; compartment_*J5p;
  IP3_Phosphorylation: IP3 => ; compartment_*J3k;
  Ca_Influx:  => Ca; compartment_*Jdelta;
