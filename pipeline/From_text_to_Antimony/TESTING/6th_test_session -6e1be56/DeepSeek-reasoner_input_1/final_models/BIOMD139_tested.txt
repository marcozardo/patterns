  compartment cytoplasm, nucleus;

  species NFkB_IkBa in cytoplasm, NFkB_IkBb in cytoplasm, NFkB_IkBe in cytoplasm;

  //// Assignment Rules:

  //// Rate Rules:

  //// Reactions:
  IKK_activation: => IKK; cytoplasm*(k_act*TNF - k_inact*IKK);
  IkBa_synthesis: => IkBa; cytoplasm*(k_syn_a*(NFkB_n^h_a/(K_a^h_a + NFkB_n^h_a)));
  IkBb_synthesis: => IkBb; cytoplasm*(k_syn_b*(NFkB_n^h_b/(K_b^h_b + NFkB_n^h_b)));
  IkBe_synthesis: => IkBe; cytoplasm*(k_syn_e*(NFkB_n^h_e/(K_e^h_e + NFkB_n^h_e)));
  IkBa_degradation: IkBa => ; cytoplasm*k_deg_a*IkBa;
  IkBb_degradation: IkBb => ; cytoplasm*k_deg_b*IkBb;
  IkBe_degradation: IkBe => ; cytoplasm*k_deg_e*IkBe;
  binding_IkBa: NFkB_c + IkBa -> NFkB_IkBa; cytoplasm*(k_on_a*NFkB_c*IkBa - k_off_a*NFkB_IkBa);
  binding_IkBb: NFkB_c + IkBb -> NFkB_IkBb; cytoplasm*(k_on_b*NFkB_c*IkBb - k_off_b*NFkB_IkBb);
  binding_IkBe: NFkB_c + IkBe -> NFkB_IkBe; cytoplasm*(k_on_e*NFkB_c*IkBe - k_off_e*NFkB_IkBe);
  degradation_bound_IkBa: NFkB_IkBa => NFkB_c; cytoplasm*k_phos_a*IKK*NFkB_IkBa;
  degradation_bound_IkBb: NFkB_IkBb => NFkB_c; cytoplasm*k_phos_b*IKK*NFkB_IkBb;
  degradation_bound_IkBe: NFkB_IkBe => NFkB_c; cytoplasm*k_phos_e*IKK*NFkB_IkBe;
  translocation_in: NFkB_c -> NFkB_n; cytoplasm*k_in*NFkB_c;
  translocation_out: NFkB_n -> NFkB_c; nucleus*k_out*NFkB_n;

  //// Events:
  TNF_pulse: at(time >= pulse_start): TNF = 1;
  TNF_end: at(time >= pulse_end): TNF = 0;
