model *Schmierer2010_HIF_ARD_Model()

  //// Compartments and Species:
  compartment Cell;
  species H_tot in Cell, H in Cell, A in Cell;
  species $H_OH in Cell, $A_OH in Cell, $A_tot in Cell, $P_tot in Cell, $F_tot in Cell, $O2 in Cell;

  //// Assignment Rules:
  H_P := 0.5*(H_tot - P_tot - K_D_P + sqrt((H_tot - P_tot - K_D_P)^2 + 4*H_tot*K_D_P));
  K_i_FH := K_D_FH * (1 + A/K_D_FA + gamma_ * A_OH/K_D_FA);
  H_F := 0.5*(H - F_tot - K_i_FH + sqrt((K_i_FH + F_tot - H)^2 + 4*K_i_FH*H));
  v_P := k_cat_P * P_tot * (1/(K_D_P + H_P + P_tot)) * (O2/(1 + O2));
  v_FH := k_cat_FH * F_tot * (1/(K_i_FH + H_F + F_tot)) * (O2/(alpha + O2));
  v_FA := (k_cat_FA * F_tot) / (K_D_FA * (1 + H_F/K_D_FH) + A + gamma_*A_OH) * (O2/(alpha + O2));
  H_OH := H_tot - H;
  A_OH := A_tot - A;

  //// Rate Rules:
  H_tot' = 1 - H_tot*(1 + v_P);
  H' = 1 - H*(1 + v_P + v_FH);
  A' = (A_tot - A*(1 + epsilon * v_FA)) / epsilon;

  //// Species initializations:
  H_tot = 0;
  H = 0;
  A = 0;

  //// Compartment initializations:
  Cell = 1;

  //// Variable initializations:
  A_tot = 100;
  P_tot = 0.2;
  F_tot = 1;
  O2 = 1;
  K_D_P = 1;
  k_cat_P = 500;
  K_D_FH = 1;
  k_cat_FH = 500;
  alpha = 0.33;
  K_D_FA = 1;
  k_cat_FA = 500;
  gamma_ = 0;
  epsilon = 1;

  //// Other declarations:
  var H_OH, A_OH, H_P, K_i_FH, H_F, v_P, v_FH, v_FA;
  const A_tot, P_tot, F_tot, O2, K_D_P, k_cat_P, K_D_FH, k_cat_FH, alpha, K_D_FA, k_cat_FA, gamma_, epsilon;

  //// Display Names:
  Cell is "Intracellular compartment (non-dimensional)";
  H_tot is "Total HIF? (non-dimensional)";
  H is "Non-CAD-hydroxylated HIF? (non-dimensional)";
  H_OH is "CAD-hydroxylated HIF? (non-dimensional)";
  A is "Unhydroxylated FIH target ankyrin repeats (non-dimensional)";
  A_OH is "Hydroxylated FIH target ankyrin repeats (non-dimensional)";
  A_tot is "Total FIH target ankyrin repeats (non-dimensional)";
  P_tot is "Total PHD concentration (non-dimensional)";
  F_tot is "Total FIH concentration (non-dimensional)";
  O2 is "Oxygen concentration (relative to K_M of PHD)";
  K_D_P is "Dissociation constant of PHD/HIF? binding (non-dimensional)";
  k_cat_P is "Catalytic rate constant for PHD (non-dimensional)";
  K_D_FH is "Dissociation constant of FIH/HIF? binding (non-dimensional)";
  k_cat_FH is "Catalytic rate constant for FIH on HIF? (non-dimensional)";
  alpha is "Ratio of K_M of FIH to K_M of PHD for oxygen";
  K_D_FA is "Dissociation constant of FIH/unhydroxylated AR binding (non-dimensional)";
  k_cat_FA is "Catalytic rate constant for FIH on ARs (non-dimensional)";
  gamma_ is "FIH affinity for hydroxylated relative to unhydroxylated ARs";
  epsilon is "Ratio of basal HIF? degradation rate constant to AR degradation rate constant";
  v_P is "Rate of PHD-dependent HIF? ODD-hydroxylation";
  v_FH is "Rate of FIH-dependent HIF? CAD-hydroxylation";
  v_FA is "Rate of FIH-dependent AR-hydroxylation";
  H_P is "Free HIF? available for PHD binding (non-dimensional)";
  H_F is "Free non-CAD-hydroxylated HIF? available for FIH binding (non-dimensional)";
  K_i_FH is "Apparent dissociation constant for FIH/HIF? binding in presence of ARs";

end
