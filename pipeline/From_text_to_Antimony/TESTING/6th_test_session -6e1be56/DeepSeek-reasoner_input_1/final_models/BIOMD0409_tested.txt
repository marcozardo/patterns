  compartment cell;

  species $MEN in cell;

  species Cdc15P in cell;
  species Cdh1 in cell;
  species Cdh1P in cell;

  //// Assignment Rules:
  APC_Cdh1 := Cdh1P;
  MEN := Cdc15;

  //// Rate Rules:

  //// Reactions:
  R1: Securin -> ; cell*k_deg_sec*APC_Cdc20*Securin;

  R7: Cdc14_inactive -> Cdc14_active; cell*k_release_cdc14*Net1P*Cdc14_inactive;
  R8: Cdc15 -> Cdc15P; cell*k_phosph_cdc15*Clb2*Cdc15;
  R9: Cdc15P -> Cdc15; cell*k_dephosph_cdc15*Cdc14_active*Cdc15P;
  R10: Cdh1 -> Cdh1P; cell*k_phosph_cdh1*Clb2*Cdh1;
  R11: Cdh1P -> Cdh1; cell*k_dephosph_cdh1*Cdc14_active*Cdh1P;
  R12: Clb2 -> ; cell*k_deg_clb2*APC_Cdh1*Clb2;
  R13: Net1 -> Net1P; cell*k_phosph_net1_men*MEN*Net1;

  //// Events:
  E1: at time >= 0: APC_Cdc20 = 1;

  //// Species initializations:
  APC_Cdc20 = 0;
  Clb2 = 100;
  Net1 = 90;
  Net1P = 10;

  Securin = 100;
  Cdc15 = 20;
  Cdc15P = 80;
  Cdh1 = 20;
  Cdh1P = 80;

  //// Compartment initializations:
  cell = 1;

  //// Variable initializations:
  k_deg_sec = 0.1;
  k_act_sep = 0.05;
  k_inact_pp2a = 0.1;
  k_phosph_net1_cdk = 0.01;
  k_dephosph_net1_pp2a = 0.1;
  k_dephosph_net1_cdc14 = 0.01;
  k_release_cdc14 = 0.05;
  k_phosph_cdc15 = 0.01;
  k_dephosph_cdc15 = 0.01;
  k_phosph_cdh1 = 0.01;
  k_dephosph_cdh1 = 0.01;
  k_deg_clb2 = 0.05;
  k_phosph_net1_men = 0.01;

  //// Other declarations:
  const cell;
  var APC_Cdc20, APC_Cdh1, MEN;

  //// Unit definitions:
  unit substance = 1e-9 mole;
  unit time_unit = 60 second;

  //// Display Names:
  cell is "Yeast cell";
  APC_Cdc20 is "APC-Cdc20 active";
  APC_Cdh1 is "APC-Cdh1 active";
  Clb2 is "Cyclin B (Clb2)";
  MEN is "Mitotic Exit Network activity";
