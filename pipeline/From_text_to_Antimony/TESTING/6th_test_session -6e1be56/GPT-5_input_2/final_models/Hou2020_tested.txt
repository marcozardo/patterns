model *Hou2020_Wuhan_SEIR()

  //// Compartments and Species:
  compartment Wuhan;
  species S in Wuhan, E in Wuhan, I in Wuhan, R in Wuhan;

  //// Assignment Rules:
  N := S + E + I + R;
  alpha2 := 0.5*alpha1;

  //// Reactions:
  Infect_by_I: S => E; (r1*alpha1*S*I)/N;
  Infect_by_E: S => E; (r2*alpha2*S*E)/N;
  Incubation: E => I; sigma*E;
  Recovery: I => R; gamma_*I;

  //// Species initializations:
  S = 11080778;
  E = 193;
  I = 27;
  R = 2;

  //// Compartment initializations:
  Wuhan = 1;

  //// Variable initializations:
  r1 = 6;
  r2 = 18;
  alpha1 = 0.4;
  sigma = 0.14;
  gamma_ = 0.048;

  //// Other declarations:
  var N, alpha2;
  const Wuhan, r1, r2, alpha1, sigma, gamma_;

  //// Display Names:
  Wuhan is "Wuhan population compartment";
  S is "Susceptible";
  E is "Exposed (latent, contagious)";
  I is "Infectious";
  R is "Recovered/Removed";
  Infect_by_I is "Infection by symptomatic infected";
  Infect_by_E is "Infection by exposed (latent)";
  Incubation is "Latent to infectious";
  Recovery is "Recovery";
end
