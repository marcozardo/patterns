  D2_form: STAT1c + STAT2c => pSTAT1pSTAT2c; Cyt*k2*aRecIFN*STAT1c*STAT2c;
  ISGF3_form: pSTAT1pSTAT2c + IRF9c => ISGF3c; Cyt*k3*pSTAT1pSTAT2c*IRF9c;

  ISGF3_exp: ISGF3n => ISGF3c; Nuc*kexp_ISGF3*ISGF3n;

  # Complex dissociation in nucleus

  D2_diss: pSTAT1pSTAT2n => STAT1n + STAT2n; Nuc*kdiss_het*pSTAT1pSTAT2n;
  ISGF3_diss: ISGF3n => pSTAT1pSTAT2n + IRF9n; Nuc*kdiss_ISGF3*ISGF3n;

  # Constitutive shuttling of monomers
  STAT1_c2n: STAT1c => STAT1n; Cyt*kimp_STAT1*STAT1c;
  STAT1_n2c: STAT1n => STAT1c; Nuc*kexp_STAT1*STAT1n;
  STAT2_c2n: STAT2c => STAT2n; Cyt*kimp_STAT2*STAT2c;
  STAT2_n2c: STAT2n => STAT2c; Nuc*kexp_STAT2*STAT2n;
  IRF9_c2n: IRF9c => IRF9n; Cyt*kimp_IRF9*IRF9c;
  IRF9_n2c: IRF9n => IRF9c; Nuc*kexp_IRF9*IRF9n;

  # Transcription (ISGF3n dominant; pSTAT1pSTAT2n contributory; SOCS3 is GAS-driven by pSTAT1dimn)
  Tx_STAT1:  => STAT1_mRNA; Nuc*(Vmax_STAT1*(ISGF3n/(K_STAT1 + ISGF3n) + w_STAT1*pSTAT1pSTAT2n/(Kp_STAT1 + pSTAT1pSTAT2n)) + Vb_STAT1);
  Tx_STAT2:  => STAT2_mRNA; Nuc*(Vmax_STAT2*(ISGF3n/(K_STAT2 + ISGF3n) + w_STAT2*pSTAT1pSTAT2n/(Kp_STAT2 + pSTAT1pSTAT2n)) + Vb_STAT2);
  Tx_IRF9:   => IRF9_mRNA;  Nuc*(Vmax_IRF9*(ISGF3n/(K_IRF9 + ISGF3n) + w_IRF9*pSTAT1pSTAT2n/(Kp_IRF9 + pSTAT1pSTAT2n)) + Vb_IRF9);
  Tx_USP18:  => USP18_mRNA; Nuc*(Vmax_USP18*(ISGF3n/(K_USP18 + ISGF3n) + w_USP18*pSTAT1pSTAT2n/(Kp_USP18 + pSTAT1pSTAT2n)) + Vb_USP18);
  Tx_SOCS1:  => SOCS1_mRNA; Nuc*(Vmax_SOCS1*(ISGF3n/(K_SOCS1 + ISGF3n) + w_SOCS1*pSTAT1pSTAT2n/(Kp_SOCS1 + pSTAT1pSTAT2n)) + Vb_SOCS1);
  Tx_IRF2:   => IRF2_mRNA;  Nuc*(Vmax_IRF2*(ISGF3n/(K_IRF2 + ISGF3n) + w_IRF2*pSTAT1pSTAT2n/(Kp_IRF2 + pSTAT1pSTAT2n)) + Vb_IRF2);
  Tx_SOCS3:  => SOCS3_mRNA; Nuc*(Vmax_SOCS3*(pSTAT1dimn/(K_SOCS3 + pSTAT1dimn)) + Vb_SOCS3);

  # mRNA degradation
  d_STAT1_m: STAT1_mRNA => ; Nuc*kdeg_m_STAT1*STAT1_mRNA;
  d_STAT2_m: STAT2_mRNA => ; Nuc*kdeg_m_STAT2*STAT2_mRNA;
  d_IRF9_m:  IRF9_mRNA  => ; Nuc*kdeg_m_IRF9*IRF9_mRNA;
  d_USP18_m: USP18_mRNA => ; Nuc*kdeg_m_USP18*USP18_mRNA;
  d_SOCS1_m: SOCS1_mRNA => ; Nuc*kdeg_m_SOCS1*SOCS1_mRNA;
  d_IRF2_m:  IRF2_mRNA  => ; Nuc*kdeg_m_IRF2*IRF2_mRNA;
  d_SOCS3_m: SOCS3_mRNA => ; Nuc*kdeg_m_SOCS3*SOCS3_mRNA;

  # Translation to proteins (cytoplasm)
  Tl_STAT1: STAT1_mRNA => STAT1c; Cyt*ktrans_STAT1*STAT1_mRNA;
  Tl_STAT2: STAT2_mRNA => STAT2c; Cyt*ktrans_STAT2*STAT2_mRNA;
  Tl_IRF9:  IRF9_mRNA  => IRF9c;  Cyt*ktrans_IRF9*IRF9_mRNA;
  Tl_USP18: USP18_mRNA => USP18;  Cyt*ktrans_USP18*USP18_mRNA;

  Tl_IRF2:  IRF2_mRNA  => IRF2;   Cyt*ktrans_IRF2*IRF2_mRNA;

  d_STAT2_p: STAT2c => ; Cyt*kdeg_p_STAT2*STAT2c;
  d_IRF9_p:  IRF9c  => ; Cyt*kdeg_p_IRF9*IRF9c;
  d_USP18_p: USP18  => ; Cyt*kdeg_p_USP18*USP18;
  d_SOCS1_p: SOCS1  => ; Cyt*kdeg_p_SOCS1*SOCS1;
  d_SOCS3_p: SOCS3  => ; Cyt*kdeg_p_SOCS3*SOCS3;
  d_IRF2_p:  IRF2   => ; Cyt*kdeg_p_IRF2*IRF2;

  E_stim:    at (time >= t_stim): IFN = IFN_stim;

/* Notes and mapping:
- Domain: molecular/cellular signal transduction in hepatocytes (type I IFN?JAK/STAT).
- Mechanistic coverage:
  * IFN binds/activates a lumped receptor (Rec -> aRecIFN).
  * aRecIFN catalyzes STAT phosphorylation and dimerization (pSTAT1 homodimer; pSTAT1:pSTAT2 heterodimer).
  * IRF9 binds pSTAT1:pSTAT2 to form ISGF3; complexes shuttle to nucleus and dissociate.
  * Nuclear pSTAT1 homodimer (GAF) drives SOCS3 mRNA (GAS-dependent).
  * Nuclear ISGF3 (dominant) and pSTAT1:pSTAT2 (weaker) drive STAT1/STAT2/IRF9/USP18/SOCS1/IRF2 mRNAs (ISRE+GAS promoters).
  * Translation to proteins; degradation of RNAs/proteins.
  * Negative regulation at receptor level by SOCS1, USP18 (and their cooperation) and SOCS3 (transient).
  * SOCS1-mediated degradation of active receptor included.
- Prestimulation/stimulation protocol:
  * Events set IFN = IFN_prestim at t_pre (default 0 h) and IFN = IFN_stim at t_stim (default 24 h).
  * IFN_prestim and IFN_stim are qualitative placeholders reflecting ~2.8 pM and 1400 pM, respectively (relative units).
- This is a reduced executable encoding consistent with the article?s described model structure (Fig. 2 and text), using mass-action and saturable transcription terms.
- Parameter values are generic placeholders to ensure simulation; article-specific calibrated values were not provided in the excerpt. */
