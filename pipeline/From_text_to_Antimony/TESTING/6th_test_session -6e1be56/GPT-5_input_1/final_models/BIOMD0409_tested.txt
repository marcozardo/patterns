  R_men_act_bfa1: MEN_inact => MEN_act; k_men_bfa1*Bfa1_P*MEN_inact;
  R_men_deact:    MEN_act   => MEN_inact; k_men_off*PP2Aact*MEN_act;

  # Bfa1 phosphorylation promoted when PP2A^Cdc55 is downregulated; dephosphorylation by PP2A and Cdc14
  R_bfa1_phos:    Bfa1_U => Bfa1_P; k_bfa1_phos*PP2Ainact*Bfa1_U;
  R_bfa1_deph_pp2a: Bfa1_P => Bfa1_U; k_bfa1_deph_pp2a*PP2Aact*Bfa1_P;
  R_bfa1_deph_c14:  Bfa1_P => Bfa1_U; k_bfa1_deph_cdc14*Cdc14_active*Bfa1_P;

  # Cdh1 activation by Cdc14 and inactivation by Cdk (Clb2-Cdk)

  R_cdh1_inact: Cdh1_A => Cdh1_U; k_cdh1_off*Clb2*Cdh1_A;

  # Clb2 degradation by APC^Cdc20 and APC^Cdh1

  R_clb2_deg_cdh1: Clb2 => ; k_clb2_cdh1*Cdh1_A*Clb2;
