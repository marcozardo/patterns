  R8_Rec_synthesis: => Rec; k_Rec_synth;

  // Cytoplasmic STAT Phosphorylation, Dimerization and ISGF3 formation

  R11_ISGF3_formation: pSTAT1pSTAT2c + IRF9c -> ISGF3c; k_ISGF3_form * pSTAT1pSTAT2c * IRF9c;
  R12_ISGF3_dissociation: ISGF3c -> pSTAT1pSTAT2c + IRF9c; k_ISGF3_diss * ISGF3c;

  R14_pSTAT1dim_export: pSTAT1dimn -> pSTAT1dimc; k_exp_pS1d * pSTAT1dimn;
  R15_pSTAT1pSTAT2_import: pSTAT1pSTAT2c -> pSTAT1pSTAT2n; k_imp_pS12 * pSTAT1pSTAT2c;
  R16_pSTAT1pSTAT2_export: pSTAT1pSTAT2n -> pSTAT1pSTAT2c; k_exp_pS12 * pSTAT1pSTAT2n;
  R17_ISGF3_import: ISGF3c -> ISGF3n; k_imp_ISGF3 * ISGF3c;
  R18_ISGF3_export: ISGF3n -> ISGF3c; k_exp_ISGF3 * ISGF3n;
  R19_STAT1_import: STAT1c -> STAT1n; k_imp_S1 * STAT1c;
  R20_STAT1_export: STAT1n -> STAT1c; k_exp_S1 * STAT1n;
  R21_STAT2_import: STAT2c -> STAT2n; k_imp_S2 * STAT2c;
  R22_STAT2_export: STAT2n -> STAT2c; k_exp_S2 * STAT2n;

  R24_IRF9_export: IRF9n -> IRF9c; k_exp_I9 * IRF9n;

  // Nuclear STAT complex dissociation (dephosphorylation implied)
  R25_pSTAT1dimn_diss: pSTAT1dimn -> 2 STAT1n; k_dis_pS1d_n * pSTAT1dimn;
  R26_pSTAT1pSTAT2n_diss: pSTAT1pSTAT2n -> STAT1n + STAT2n; k_dis_pS12_n * pSTAT1pSTAT2n;
  R27_ISGF3n_diss: ISGF3n -> STAT1n + STAT2n + IRF9n; k_dis_ISGF3_n * ISGF3n;

  // Transcription Factor Binding to Promoters
  R28_GAS_binding: pSTAT1dimn + GASbs -> OccGASbs; k_bind_GAS * pSTAT1dimn * GASbs;
  R29_GAS_unbinding: OccGASbs -> pSTAT1dimn + GASbs; k_unbind_GAS * OccGASbs;
  R30_ISRE_binding: ISGF3n + ISREbs -> OccISREbs; k_bind_ISRE * ISGF3n * ISREbs;
  R31_ISRE_unbinding: OccISREbs -> ISGF3n + ISREbs; k_unbind_ISRE * OccISREbs;
  R32_GAS_ISRE_binding_pS1S2: pSTAT1pSTAT2n + GAS_ISREbs -> OccGAS_ISREbs_pS1S2n; k_bind_GI_pS12 * pSTAT1pSTAT2n * GAS_ISREbs;
  R33_GAS_ISRE_unbinding_pS1S2: OccGAS_ISREbs_pS1S2n -> pSTAT1pSTAT2n + GAS_ISREbs; k_unbind_GI_pS12 * OccGAS_ISREbs_pS1S2n;
  R34_GAS_ISRE_binding_ISGF3: ISGF3n + GAS_ISREbs -> OccGAS_ISREbs_ISGF3n; k_bind_GI_ISGF3 * ISGF3n * GAS_ISREbs;

  R42_IRF2_transcription: => IRF2mRNA; nucleus * (Vmax_IRF2_pS12 * OccGAS_ISREbs_pS1S2n + Vmax_IRF2_ISGF3 * OccGAS_ISREbs_ISGF3n) / (Km_IRF2 + OccGAS_ISREbs_pS1S2n + OccGAS_ISREbs_ISGF3n);
  // Note: ISRE-only genes (DDX58, etc.) are omitted for model simplicity, but would follow the SOCS3 transcription model with OccISREbs.
