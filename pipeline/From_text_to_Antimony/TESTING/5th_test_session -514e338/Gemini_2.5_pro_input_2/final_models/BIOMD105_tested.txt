model *Proctor2007_UbiquitinProteasome()

  //// Compartments and Species:
  compartment cell;
  species NatP in cell, MisP in cell, Ub in cell, E1 in cell, E2 in cell;
  species E3 in cell, DUB in cell, Proteasome in cell, E3_MisP in cell, E1_Ub in cell;
  species E2_Ub in cell, MisP_Ub in cell, MisP_Ub2 in cell, MisP_Ub3 in cell, MisP_Ub4 in cell;
  species MisP_Ub5 in cell, MisP_Ub6 in cell, MisP_Ub7 in cell, MisP_Ub8 in cell;
  species MisP_Ub4_Proteasome in cell, MisP_Ub5_Proteasome in cell, MisP_Ub6_Proteasome in cell;
  species MisP_Ub7_Proteasome in cell, MisP_Ub8_Proteasome in cell, AggP in cell, SeqAggP in cell;
  species AggP_Proteasome in cell, degUb4 in cell, degUb5 in cell, degUb6 in cell, degUb7 in cell;
  species degUb8 in cell;
  species $ROS in cell, $ATP in cell, $ADP in cell, $AMP in cell;

  //// Reactions:
  # Protein synthesis, misfolding, and refolding
  Protein_synthesis:         => NatP; k_synth;
  Misfolding:         NatP + ROS -> MisP + ROS; k_misfold*NatP*ROS;
  Refolding:          MisP       => NatP; k_refold*MisP;

  # MisP recognition by E3 ligase
  MisP_E3_binding:    E3 + MisP  => E3_MisP; k_E3_bind*E3*MisP;
  MisP_E3_release:    E3_MisP    => E3 + MisP; k_E3_release*E3_MisP;

  # Ubiquitin activation and transfer
  E1_Ub_binding:      E1 + Ub + ATP -> E1_Ub + AMP; k_E1_bind*E1*Ub*ATP/(5000 + ATP);
  E2_Ub_binding:      E2 + E1_Ub    => E1 + E2_Ub; k_E2_bind*E2*E1_Ub;
  Monoubiquitination: E2_Ub + E3_MisP => E2 + E3 + MisP_Ub; k_monoUb*E2_Ub*E3_MisP;

  # Polyubiquitination chain elongation
  Polyubiquitination_1: E2_Ub + MisP_Ub   => MisP_Ub2 + E2; k_polyUb*E2_Ub*MisP_Ub;
  Polyubiquitination_2: E2_Ub + MisP_Ub2  => MisP_Ub3 + E2; k_polyUb*E2_Ub*MisP_Ub2;
  Polyubiquitination_3: E2_Ub + MisP_Ub3  => MisP_Ub4 + E2; k_polyUb*E2_Ub*MisP_Ub3;
  Polyubiquitination_4: E2_Ub + MisP_Ub4  => MisP_Ub5 + E2; k_polyUb*E2_Ub*MisP_Ub4;
  Polyubiquitination_5: E2_Ub + MisP_Ub5  => MisP_Ub6 + E2; k_polyUb*E2_Ub*MisP_Ub5;
  Polyubiquitination_6: E2_Ub + MisP_Ub6  => MisP_Ub7 + E2; k_polyUb*E2_Ub*MisP_Ub6;
  Polyubiquitination_7: E2_Ub + MisP_Ub7  => MisP_Ub8 + E2; k_polyUb*E2_Ub*MisP_Ub7;

  # Deubiquitination of free substrates by DUBs
  Deubiquitination_8: DUB + MisP_Ub8 => DUB + MisP_Ub7 + Ub; k_deUb*DUB*MisP_Ub8;
  Deubiquitination_7: DUB + MisP_Ub7 => DUB + MisP_Ub6 + Ub; k_deUb*DUB*MisP_Ub7;
  Deubiquitination_6: DUB + MisP_Ub6 => DUB + MisP_Ub5 + Ub; k_deUb*DUB*MisP_Ub6;
  Deubiquitination_5: DUB + MisP_Ub5 => DUB + MisP_Ub4 + Ub; k_deUb*DUB*MisP_Ub5;
  Deubiquitination_4: DUB + MisP_Ub4 => DUB + MisP_Ub3 + Ub; k_deUb*DUB*MisP_Ub4;
  Deubiquitination_3: DUB + MisP_Ub3 => DUB + MisP_Ub2 + Ub; k_deUb*DUB*MisP_Ub3;
  Deubiquitination_2: DUB + MisP_Ub2 => DUB + MisP_Ub + Ub; k_deUb*DUB*MisP_Ub2;
  Deubiquitination_1: DUB + MisP_Ub  => DUB + MisP + Ub; k_deUb*DUB*MisP_Ub;

  # Binding of polyubiquitinated substrates to the proteasome (chain length >= 4)
  Proteasome_binding_4: MisP_Ub4 + Proteasome => MisP_Ub4_Proteasome; k_prot_bind*MisP_Ub4*Proteasome;
  Proteasome_binding_5: MisP_Ub5 + Proteasome => MisP_Ub5_Proteasome; k_prot_bind*MisP_Ub5*Proteasome;
  Proteasome_binding_6: MisP_Ub6 + Proteasome => MisP_Ub6_Proteasome; k_prot_bind*MisP_Ub6*Proteasome;
  Proteasome_binding_7: MisP_Ub7 + Proteasome => MisP_Ub7_Proteasome; k_prot_bind*MisP_Ub7*Proteasome;
  Proteasome_binding_8: MisP_Ub8 + Proteasome => MisP_Ub8_Proteasome; k_prot_bind*MisP_Ub8*Proteasome;

  # Deubiquitination of proteasome-bound substrates by DUBs
  Deubiquitination_Bound_8: DUB + MisP_Ub8_Proteasome => DUB + MisP_Ub7_Proteasome + Ub; k_deUb_bound*DUB*MisP_Ub8_Proteasome;
  Deubiquitination_Bound_7: DUB + MisP_Ub7_Proteasome => DUB + MisP_Ub6_Proteasome + Ub; k_deUb_bound*DUB*MisP_Ub7_Proteasome;
  Deubiquitination_Bound_6: DUB + MisP_Ub6_Proteasome => DUB + MisP_Ub5_Proteasome + Ub; k_deUb_bound*DUB*MisP_Ub6_Proteasome;
  Deubiquitination_Bound_5: DUB + MisP_Ub5_Proteasome => DUB + MisP_Ub4_Proteasome + Ub; k_deUb_bound*DUB*MisP_Ub5_Proteasome;
  Dissociation_from_Proteasome: DUB + MisP_Ub4_Proteasome => DUB + MisP_Ub3 + Ub + Proteasome; k_deUb_bound*DUB*MisP_Ub4_Proteasome;

  # Degradation by the proteasome
  Proteasome_Activity_4: MisP_Ub4_Proteasome + ATP => Proteasome + 4 Ub + ADP + degUb4; k_degrade*MisP_Ub4_Proteasome*ATP/(5000 + ATP);
  Proteasome_Activity_5: MisP_Ub5_Proteasome + ATP => Proteasome + 5 Ub + ADP + degUb5; k_degrade*MisP_Ub5_Proteasome*ATP/(5000 + ATP);
  Proteasome_Activity_6: MisP_Ub6_Proteasome + ATP => Proteasome + 6 Ub + ADP + degUb6; k_degrade*MisP_Ub6_Proteasome*ATP/(5000 + ATP);
  Proteasome_Activity_7: MisP_Ub7_Proteasome + ATP => Proteasome + 7 Ub + ADP + degUb7; k_degrade*MisP_Ub7_Proteasome*ATP/(5000 + ATP);
  Proteasome_Activity_8: MisP_Ub8_Proteasome + ATP => Proteasome + 8 Ub + ADP + degUb8; k_degrade*MisP_Ub8_Proteasome*ATP/(5000 + ATP);

  # Protein aggregation and proteasome inhibition
  Aggregation1:             2 MisP          => AggP; k_agg1*MisP*(MisP-1)/2;
  Aggregation2:             MisP + AggP     => 2 AggP; k_agg1*MisP*AggP;
  Aggregation3:             2 MisP_Ub       => AggP; k_agg2*MisP_Ub*(MisP_Ub-1)/2;
  Aggregation_ub_growth:    MisP_Ub + AggP  => 2 AggP; k_agg2*MisP_Ub*AggP; # Representative of MisP_Ubi + AggP
  Aggregation10:            2 MisP_Ub8      => AggP; k_agg2*MisP_Ub8*(MisP_Ub8-1)/2;
  Aggregation11:            MisP + MisP_Ub  => AggP; k_agg2*MisP*MisP_Ub;
  Aggregation54:            MisP_Ub7 + MisP_Ub8 => AggP; k_agg2*MisP_Ub7*MisP_Ub8;
  SequesteringOfAggregates: AggP            => SeqAggP; k_seq*AggP;
  ProteasomeInhibition:     AggP + Proteasome => AggP_Proteasome; k_prot_inhibit*AggP*Proteasome;

  //// Events:
  E1_inactivation: at time >= 1800: k_E1_bind = 0;

  //// Species initializations:
  NatP = 500;
  MisP = 0;
  Ub = 500;
  E1 = 100;
  E2 = 100;
  E3 = 100;
  DUB = 200;
  Proteasome = 100;
  ROS = 10;
  ATP = 10000;
  ADP = 1000;
  AMP = 1000;
  E3_MisP = 0;
  E1_Ub = 0;
  E2_Ub = 0;
  MisP_Ub = 0;
  MisP_Ub2 = 0;
  MisP_Ub3 = 0;
  MisP_Ub4 = 0;
  MisP_Ub5 = 0;
  MisP_Ub6 = 0;
  MisP_Ub7 = 0;
  MisP_Ub8 = 0;
  MisP_Ub4_Proteasome = 0;
  MisP_Ub5_Proteasome = 0;
  MisP_Ub6_Proteasome = 0;
  MisP_Ub7_Proteasome = 0;
  MisP_Ub8_Proteasome = 0;
  AggP = 0;
  SeqAggP = 0;
  AggP_Proteasome = 0;
  degUb4 = 0;
  degUb5 = 0;
  degUb6 = 0;
  degUb7 = 0;
  degUb8 = 0;

  //// Compartment initializations:
  cell = 1;

  //// Variable initializations:
  k_synth = 0.01;            # per second
  k_misfold = 2.0E-6;        # per molecule per second
  k_refold = 4.0E-6;         # per second
  k_E3_bind = 1.7E-5;        # per molecule per second
  k_E3_release = 2.0E-4;     # per second
  k_E1_bind = 2.0E-4;        # per molecule per second
  k_E2_bind = 1.0E-3;        # per molecule per second
  k_monoUb = 1.0E-3;         # per molecule per second
  k_polyUb = 1.0E-2;         # per molecule per second
  k_deUb = 1.0E-5;           # per molecule per second
  k_prot_bind = 1.0E-5;      # per molecule per second
  k_deUb_bound = 1.0E-5;     # per molecule per second
  k_degrade = 1.0E-3;        # per second
  k_agg1 = 1.0E-8;           # per molecule per second
  k_agg2 = 1.0E-8;           # per molecule per second
  k_seq = 1.0E-3;            # per second
  k_prot_inhibit = 1.0E-5;   # per molecule per second

  //// Other declarations:
  var k_E1_bind; # Variable because it is changed by an event
  const cell, k_synth, k_misfold, k_refold, k_E3_bind, k_E3_release, k_E2_bind;
  const k_monoUb, k_polyUb, k_deUb, k_prot_bind, k_deUb_bound, k_degrade, k_agg1;
  const k_agg2, k_seq, k_prot_inhibit;

end
