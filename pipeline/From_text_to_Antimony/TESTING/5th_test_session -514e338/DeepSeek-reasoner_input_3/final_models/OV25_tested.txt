model *Minimal_Tumor_Immune_Model()

  //// Compartments and Species:
  compartment tissue;
  species U in tissue, I in tissue, V in tissue, T in tissue, D in tissue;

  //// Assignment Rules:
  N := U + I + T;  # Total cell volume

  //// Rate Rules:
  # No rate rules - all dynamics governed by reactions

  //// Reactions:
  # Tumor growth
  tumor_growth: => U; r*U;
  
  # Virus infection
  infection: U -> I; beta*(U*V)/N;
  
  # Infected cell lysis
  lysis: I => ; delta_I*I;
  
  # Virus production from infected cells
  virus_release: I => V; alpha*delta_I*I;
  
  # Virus decay
  virus_decay: V => ; delta_V*V;
  
  # T cell-mediated killing of uninfected cells
  kill_U: U => ; (kappa0 + c_kill*I)*(U*T)/N;
  
  # T cell-mediated killing of infected cells  
  kill_I: I => ; (kappa0 + c_kill*I)*(I*T)/N;
  
  # T cell activation by cytokines
  T_activation: => T; c_T*I;
  
  # T cell activation by dendritic cells
  DC_activation: => T; chi_D*D;
  
  # T cell decay
  T_decay: T => ; delta_T*T;
  
  # Dendritic cell decay
  DC_decay: D => ; delta_D*D;

  //// Events:
  # Virus injections (example: days 0, 2, 4)
  at(time >= 0): V = V + virus_dose;
  at(time >= 2): V = V + virus_dose;
  at(time >= 4): V = V + virus_dose;
  
  # Dendritic cell injections (example: days 1, 3, 5)
  at(time >= 1): D = D + DC_dose;
  at(time >= 3): D = D + DC_dose;
  at(time >= 5): D = D + DC_dose;

  //// Species initializations:
  U = 100;   # Initial uninfected tumor volume (mm?)
  I = 0;     # Initial infected tumor volume
  V = 0;     # Initial virus concentration
  T = 0;     # Initial T cell concentration
  D = 0;     # Initial dendritic cell concentration

  //// Compartment initializations:
  tissue = 1;

  //// Variable initializations:
  # Core parameters from Table 2
  r = 0.3198;           # Tumor growth rate (day??)
  beta = 0.00100854;    # Infection rate parameter
  c_T = 1.428064;       # T cell activation rate by cytokines
  c_kill = 6.234e-7;    # Enhanced killing rate parameter
  delta_D = 4.901894;   # Dendritic cell decay rate (day??)
  
  # Additional parameters (values not specified in text)
  kappa0 = 0.1;         # Baseline killing rate (placeholder)
  delta_I = 1.0;        # Infected cell lysis rate (placeholder)
  alpha = 100;          # Virions released per cell (placeholder)
  delta_V = 10;         # Virus decay rate (placeholder)
  chi_D = 0.1;          # T cell activation by DCs (placeholder)
  delta_T = 0.5;        # T cell decay rate (placeholder)
  
  # Treatment doses (placeholders)
  virus_dose = 5e9;     # Virus injection amount
  DC_dose = 1e6;        # DC injection amount

  //// Other declarations:
  const r, beta, kappa0, c_kill, delta_I, alpha, delta_V, c_T, chi_D, delta_T, delta_D, virus_dose, DC_dose;
  var N;

end
