```antimony
model *Kok2020_IFNa_Signaling()

  //// Compartments and Species:
  compartment Cytoplasm, Nucleus;
  species $IFN; // External IFN, constant or changed by events
  species Rec in Cytoplasm, aRecIFN in Cytoplasm;
  species STAT1c in Cytoplasm, STAT2c in Cytoplasm, IRF9c in Cytoplasm;
  species pSTAT1dimc in Cytoplasm, pSTAT1pSTAT2c in Cytoplasm, ISGF3c in Cytoplasm;
  species SOCS1 in Cytoplasm, SOCS3 in Cytoplasm, USP18 in Cytoplasm, IRF2 in Cytoplasm;
  species STAT1mRNA in Cytoplasm, STAT2mRNA in Cytoplasm, IRF9mRNA in Cytoplasm;
  species SOCS1mRNA in Cytoplasm, SOCS3mRNA in Cytoplasm, USP18mRNA in Cytoplasm, IRF2mRNA in Cytoplasm;
  species STAT1n in Nucleus, STAT2n in Nucleus, IRF9n in Nucleus;
  species pSTAT1dimn in Nucleus, pSTAT1pSTAT2n in Nucleus, ISGF3n in Nucleus;

  //// Reactions:
  // Note: All parameter values are placeholders as they are not available in the main text.
  
  // 1. Receptor dynamics
  R1: Rec + IFN -> aRecIFN; k_bind_IFN * Rec * IFN / ((1 + SOCS1/KI_SOCS1) * (1 + SOCS3/KI_SOCS3)); # Receptor activation inhibited by SOCS1/3
  R2: aRecIFN -> Rec; k_unbind_IFN * aRecIFN; # Receptor deactivation
  R3: aRecIFN ->; k_degradation_aRec_SOCS1 * aRecIFN * SOCS1; # SOCS1-mediated degradation of active receptor
  R4: aRecIFN ->; k_degradation_aRec_SOCS1_USP18 * aRecIFN * SOCS1 * USP18; # SOCS1:USP18-mediated degradation of active receptor
  R5: -> Rec; k_synth_Rec; # Receptor synthesis
  R6: Rec ->; k_degradation_Rec * Rec; # Basal receptor degradation

  // 2. Cytoplasmic signaling: STAT phosphorylation and complex formation
  R7: 2 STAT1c -> pSTAT1dimc; k_pSTAT1_dimerization * STAT1c^2 * aRecIFN / (1 + USP18/KI_USP18); # pSTAT1 homodimer formation, inhibited by USP18
  R8: STAT1c + STAT2c -> pSTAT1pSTAT2c; k_pSTAT1_pSTAT2_heterodimerization * STAT1c * STAT2c * aRecIFN / (1 + USP18/KI_USP18); # pSTAT1:pSTAT2 heterodimer formation, inhibited by USP18
  R9: pSTAT1pSTAT2c + IRF9c -> ISGF3c; k_ISGF3_formation * pSTAT1pSTAT2c * IRF9c; # ISGF3 trimer formation

  // 3. Nucleocytoplasmic transport
  R10: pSTAT1dimc -> pSTAT1dimn; k_import_pSTAT1dim * pSTAT1dimc;
  R11: pSTAT1pSTAT2c -> pSTAT1pSTAT2n; k_import_pSTAT1pSTAT2 * pSTAT1pSTAT2c;
  R12: ISGF3c -> ISGF3n; k_import_ISGF3 * ISGF3c;
  R13: STAT1c -> STAT1n; k_import_STAT1 * STAT1c;
  R14: STAT1n -> STAT1c; k_export_STAT1 * STAT1n;
  R15: STAT2c -> STAT2n; k_import_STAT2 * STAT2c;
  R16: STAT2n -> STAT2c; k_export_STAT2 * STAT2n;
  R17: IRF9c -> IRF9n; k_import_IRF9 * IRF9c;
  R18: IRF9n -> IRF9c; k_export_IRF9 * IRF9n;

  // 4. Nuclear complex dissociation
  R19: pSTAT1dimn -> 2 STAT1n; k_dissociation_pSTAT1dim * pSTAT1dimn;
  R20: pSTAT1pSTAT2n -> STAT1n + STAT2n; k_dissociation_pSTAT1pSTAT2 * pSTAT1pSTAT2n;
  R21: ISGF3n -> STAT1n + STAT2n + IRF9n; k_dissociation_ISGF3 * ISGF3n;

  // 5. Gene transcription (mRNA synthesis)
  R22: -> SOCS3mRNA; k_synth_SOCS3mRNA * pSTAT1dimn; # GAS-driven
  R23: -> STAT1mRNA; k_synth_STAT1mRNA_pSTAT1pSTAT2 * pSTAT1pSTAT2n + k_synth_STAT1mRNA_ISGF3 * ISGF3n; # GAS/ISRE-driven
  R24: -> STAT2mRNA; k_synth_STAT2mRNA_pSTAT1pSTAT2 * pSTAT1pSTAT2n + k_synth_STAT2mRNA_ISGF3 * ISGF3n; # GAS/ISRE-driven
  R25: -> IRF9mRNA; k_synth_IRF9mRNA_pSTAT1pSTAT2 * pSTAT1pSTAT2n + k_synth_IRF9mRNA_ISGF3 * ISGF3n; # GAS/ISRE-driven
  R26: -> USP18mRNA; k_synth_USP18mRNA_pSTAT1pSTAT2 * pSTAT1pSTAT2n + k_synth_USP18mRNA_ISGF3 * ISGF3n; # GAS/ISRE-driven
  R27: -> SOCS1mRNA; k_synth_SOCS1mRNA_pSTAT1pSTAT2 * pSTAT1pSTAT2n + k_synth_SOCS1mRNA_ISGF3 * ISGF3n; # GAS/ISRE-driven
  R28: -> IRF2mRNA; k_synth_IRF2mRNA_pSTAT1pSTAT2 * pSTAT1pSTAT2n + k_synth_IRF2mRNA_ISGF3 * ISGF3n; # GAS/ISRE-driven

  // 6. mRNA degradation
  R29: SOCS3mRNA ->; k_degradation_SOCS3mRNA * SOCS3mRNA;
  R30: STAT1mRNA ->; k_degradation_STAT1mRNA * STAT1mRNA;
  R31: STAT2mRNA ->; k_degradation_STAT2mRNA * STAT2mRNA;
  R32: IRF9mRNA ->; k_degradation_IRF9mRNA * IRF9mRNA;
  R33: USP18mRNA ->; k_degradation_USP18mRNA * USP18mRNA;
  R34: SOCS1mRNA ->; (k_degradation_SOCS1mRNA_basal + k_degradation_SOCS1mRNA_IRF2 * IRF2) * SOCS1mRNA; # Basal and IRF2-enhanced degradation
  R35: IRF2mRNA ->; k_degradation_IRF2mRNA * IRF2mRNA;

  // 7. Protein synthesis (translation)
  R36: -> SOCS3; k_translation_SOCS3 * SOCS3mRNA;
  R37: -> STAT1c; k_translation_STAT1 * STAT1mRNA;
  R38: -> STAT2c; k_translation_STAT2 * STAT2mRNA;
  R39: -> IRF9c; k_translation_IRF9 * IRF9mRNA;
  R40: -> USP18; k_translation_USP18 * USP18mRNA;
  R41: -> SOCS1; k_translation_SOCS1 * SOCS1mRNA;
  R42: -> IRF2; k_translation_IRF2 * IRF2mRNA;
  
  // 8. Basal protein synthesis and degradation
  R43: -> STAT1c; k_synth_STAT1_basal;
  R44: STAT1c ->; k_degradation_STAT1 * STAT1c;
  R45: STAT1n ->; k_degradation_STAT1 * STAT1n;
  R46: -> STAT2c; k_synth_STAT2_basal;
  R47: STAT2c ->; k_degradation_STAT2 * STAT2c;
  R48: STAT2n ->; k_degradation_STAT2 * STAT2n;
  R49: -> IRF9c; k_synth_IRF9_basal;
  R50: IRF9c ->; k_degradation_IRF9 * IRF9c;
  R51: IRF9n ->; k_degradation_IRF9 * IRF9n;
  R52: -> USP18; k_synth_USP18_basal;
  R53: USP18 ->; k_degradation_USP18 * USP18;
  R54: SOCS1 ->; k_degradation_SOCS1 * SOCS1;
  R55: SOCS3 ->; k_degradation_SOCS3 * SOCS3;
  R56: IRF2 ->; k_degradation_IRF2 * IRF2;

  //// Events:
  // Simulates prestimulation at t=0 and stimulation at t=24h
  E_prestim: at time >= 0: IFN = IFN_prestim;
  E_stim: at time >= 24: IFN = IFN_stim;

  //// Species initializations:
  IFN = 0; // Set by events
  // Initial protein amounts based on Fig 3C for untreated Huh7.5 cells (molecules/cell)
  STAT1c = 7.5e5;
  STAT2c = 5.0e4;
  IRF9c = 5.0e3;
  USP18 = 5.0e3;
  Rec = 1000;
  // Initialize all other species to 0 or near-zero
  aRecIFN = 0;
  pSTAT1dimc = 0; pSTAT1pSTAT2c = 0; ISGF3c = 0;
  SOCS1 = 0; SOCS3 = 0; IRF2 = 0;
  STAT1mRNA = 0; STAT2mRNA = 0; IRF9mRNA = 0;
  SOCS1mRNA = 0; SOCS3mRNA = 0; USP18mRNA = 0; IRF2mRNA = 0;
  STAT1n = 0; STAT2n = 0; IRF9n = 0;
  pSTAT1dimn = 0; pSTAT1pSTAT2n = 0; ISGF3n = 0;

  //// Compartment initializations:
  Cytoplasm = 1;
  Nucleus = 1;

  //// Variable initializations:
  // Parameters for stimulation events (in pM)
  IFN_prestim = 2.8;  // e.g., low dose prestimulation
  IFN_stim = 1400; // e.g., high dose stimulation
  // NOTE: All kinetic parameter values below are placeholders as they are not available in the main text of the article.
  // Their values are in Appendix Table S3 of the original publication.
  k_bind_IFN = 0.1; KI_SOCS1 = 1; KI_SOCS3 = 1;
  k_unbind_IFN = 0.1;
  k_degradation_aRec_SOCS1 = 0.1;
  k_degradation_aRec_SOCS1_USP18 = 0.1;
  k_synth_Rec = 10; k_degradation_Rec = 0.01;
  k_pSTAT1_dimerization = 0.1; KI_USP18 = 1;
  k_pSTAT1_pSTAT2_heterodimerization = 0.1;
  k_ISGF3_formation = 0.1;
  k_import_pSTAT1dim = 0.1; k_import_pSTAT1pSTAT2 = 0.1; k_import_ISGF3 = 0.1;
  k_import_STAT1 = 0.1; k_export_STAT1 = 0.1;
  k_import_STAT2 = 0.1; k_export_STAT2 = 0.1;
  k_import_IRF9 = 0.1; k_export_IRF9 = 0.1;
  k_dissociation_pSTAT1dim = 0.1; k_dissociation_pSTAT1pSTAT2 = 0.1; k_dissociation_ISGF3 = 0.1;
  k_synth_SOCS3mRNA = 0.1;
  k_synth_STAT1mRNA_pSTAT1pSTAT2 = 0.1; k_synth_STAT1mRNA_ISGF3 = 0.1;
  k_synth_STAT2mRNA_pSTAT1pSTAT2 = 0.1; k_synth_STAT2mRNA_ISGF3 = 0.1;
  k_synth_IRF9mRNA_pSTAT1pSTAT2 = 0.1; k_synth_IRF9mRNA_ISGF3 = 0.1;
  k_synth_USP18mRNA_pSTAT1pSTAT2 = 0.1; k_synth_USP18mRNA_ISGF3 = 0.1;
  k_synth_SOCS1mRNA_pSTAT1pSTAT2 = 0.1; k_synth_SOCS1mRNA_ISGF3 = 0.1;
  k_synth_IRF2mRNA_pSTAT1pSTAT2 = 0.1; k_synth_IRF2mRNA_ISGF3 = 0.1;
  k_degradation_SOCS3mRNA = 0.1; k_degradation_STAT1mRNA = 0.1; k_degradation_STAT2mRNA = 0.1;
  k_degradation_IRF9mRNA = 0.1; k_degradation_USP18mRNA = 0.1;
  k_degradation_SOCS1mRNA_basal = 0.1; k_degradation_SOCS1mRNA_IRF2 = 0.1;
  k_degradation_IRF2mRNA = 0.1;
  k_translation_SOCS3 = 0.1; k_translation_STAT1 = 0.1; k_translation_STAT2 = 0.1;
  k_translation_IRF9 = 0.1; k_translation_USP18 = 0.1; k_translation_SOCS1 = 0.1;
  k_translation_IRF2 = 0.1;
  // Basal synthesis rates are set to balance initial degradation
  k_degradation_STAT1 = 0.01; k_synth_STAT1_basal = 7.5e5 * 0.01;
  k_degradation_STAT2 = 0.01; k_synth_STAT2_basal = 5.0e4 * 0.01;
  k_degradation_IRF9 = 0.01; k_synth_IRF9_basal = 5.0e3 * 0.01;
  k_degradation_USP18 = 0.01; k_synth_USP18_basal = 5.0e3 * 0.01;
  k_degradation_SOCS1 = 0.01; k_degradation_SOCS3 = 0.01; k_degradation_IRF2 = 0.01;

  //// Unit definitions:
  unit substance = 1 mole;
  unit time_unit = 3600 second; // hour
  unit volume = 1 liter;
  
  //// Display Names:
  time_unit is "hour";
  IFN is "Interferon alpha";
  Rec is "IFN alpha Receptor";
  aRecIFN is "Activated IFN alpha Receptor";
  STAT1c is "Cytoplasmic STAT1";
  STAT2c is "Cytoplasmic STAT2";
  IRF9c is "Cytoplasmic IRF9";
  pSTAT1dimc is "Cytoplasmic pSTAT1 homodimer";
  pSTAT1pSTAT2c is "Cytoplasmic pSTAT1:pSTAT2 heterodimer";
  ISGF3c is "Cytoplasmic ISGF3 complex";
  SOCS1 is "Suppressor of cytokine signaling 1";
  SOCS3 is "Suppressor of cytokine signaling 3";
  USP18 is "Ubiquitin specific peptidase 18";
  IRF2 is "Interferon regulatory factor 2";
  STAT1n is "Nuclear STAT1";
  STAT2n is "Nuclear STAT2";
  IRF9n is "Nuclear IRF9";
  pSTAT1dimn is "Nuclear pSTAT1 homodimer";
  pSTAT1pSTAT2n is "Nuclear pSTAT1:pSTAT2 heterodimer";
  ISGF3n is "Nuclear ISGF3 complex";
end
```
