  Cdk := Clb2;  # proxy for Clb2-Cdk activity

  //// Rate Rules:
  MEN' = k_men_act1*Cdc14_free*(1 - MEN)/(1 + a_cdk*Cdk) + k_men_act2*(PP2A_base - PP2A)*(1 - MEN) - k_men_inh*MEN;

  //// Reactions:

  R3_bind: Net1 + Cdc14_free -> Cdc14_bound; k_bind*Net1*Cdc14_free;

  R8_clb2_degCdh1: Clb2 -> ; k_apcCdh1_clb2*Cdh1*Clb2;
