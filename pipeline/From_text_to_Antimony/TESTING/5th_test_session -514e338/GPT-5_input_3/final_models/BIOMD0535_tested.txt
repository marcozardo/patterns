  species IL6_g in GI;
  species $pSTAT3_g in GI, $pSTAT3_l in Liver;

  //// Assignment Rules:
  Sig_g := alpha_t*IL6_sIL6R_s + alpha_c*IL6_s;      # Effective IL-6 signal (trans + classical) sensed by GI

  //// Rate Rules:
  pSTAT3_g' = Vphos_g * (Sig_g/(Ksig + Sig_g)) * STAT3_free_g - Vdephos_g * (pSTAT3_g/(Kdephos + pSTAT3_g));
  pSTAT3_l' = Vphos_l * (Sig_l/(Ksig + Sig_l)) * STAT3_free_l - Vdephos_l * (pSTAT3_l/(Kdephos + pSTAT3_l));

  //// Reactions:

  R2_sequester: IL6_sIL6R_s + sgp130_s -> IL6_sIL6R_sgp130_s; kon2*IL6_sIL6R_s*sgp130_s - koff2*IL6_sIL6R_sgp130_s;

  # IL-6 production (gut), exchange with serum, and degradation
  R3_IL6_g_prod:  => IL6_g; ksyn_IL6_g;
  R4_IL6_g_deg:   IL6_g => ; kdeg_IL6*IL6_g;
  R5_IL6_g2s:     IL6_g => IL6_s; k_gi_to_ser*IL6_g;
  R6_IL6_s2g:     IL6_s => IL6_g; k_ser_to_gi*IL6_s;
  R7_IL6_s_deg:   IL6_s => ; kdeg_IL6*IL6_s;

  # sIL-6R production (basal + CRP-mediated shedding) and degradation
  R8_sIL6R_prod:  => sIL6R_s; ksyn_sIL6R + k_shed*CRP_s;
  R9_sIL6R_deg:   sIL6R_s => ; kdeg_sIL6R*sIL6R_s;

  # sgp130 production and degradation
  R10_sgp130_prod: => sgp130_s; ksyn_sgp130;
  R11_sgp130_deg:  sgp130_s => ; kdeg_sgp130*sgp130_s;

  # CRP production in liver (pSTAT3-driven) secreted to serum and degradation
  R12_CRP_prod:  => CRP_s; Vcrp_basal + Vcrp_max*(pSTAT3_l/(Kcrp + pSTAT3_l));
  R13_CRP_deg:   CRP_s => ; kdeg_CRP*CRP_s;

  # Clearance of inactive/sequestered complexes (optional lumped clearance)
  R14_cx1_clear: IL6_sIL6R_s => ; kdeg_complex*IL6_sIL6R_s;
  R15_cx2_clear: IL6_sIL6R_sgp130_s => ; kdeg_complex*IL6_sIL6R_sgp130_s;
