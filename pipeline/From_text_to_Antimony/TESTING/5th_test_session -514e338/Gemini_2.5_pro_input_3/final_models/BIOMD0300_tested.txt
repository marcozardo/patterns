model *Schmierer2010_HIF_FIH_ARD_FullModel()

  //// Compartments and Species:
  compartment cell;
  species H_hat_tot in cell, H_hat in cell, A_hat in cell;

  //// Assignment Rules:
  # The CAD-hydroxylated forms of HIFa (H_hat_OH) and ARD proteins (A_hat_OH) are defined by mass conservation.
  H_hat_OH := H_hat_tot - H_hat;
  A_hat_OH := A_hat_tot - A_hat;

  # Hydroxylation Rate Functions (dimensionless)
  # Eq 4: PHD-dependent HIFa ODD-hydroxylation
  nu_p_prime := k_cat_P_prime * P_hat_tot / (K_D_P + H_hat_p + P_hat_tot) * (O2_tilde / (1 + O2_tilde));

  # Eq 5: Intermediate term for nu_p_prime. The symbol H_hat_p is used, based on context correcting a typo in the paper (A_p).
  # The paper's description of H_hat_p as 'free HIFa' is inconsistent with the mathematical formula, which is a variant of the quadratic solution for a complex.
  # The model here directly implements the formula as written.
  H_hat_p := -0.5 * (H_hat_tot - P_hat_tot - K_D_P + sqrt((K_D_P + P_hat_tot - H_hat_tot)^2 + 4 * K_D_P * H_hat_tot));
  
  # Eq 6: FIH-dependent HIFa CAD-hydroxylation
  nu_FH_prime := k_cat_FH_prime * F_hat_tot / (K_hat_i_FH + H_hat_F + F_hat_tot) * (O2_tilde / (alpha + O2_tilde));

  # Eq 8: Apparent dissociation constant for FIH/HIFa binding, considering competition by ARD proteins.
  K_hat_i_FH := K_D_FH * (1 + A_hat/K_D_FA + gamma_*A_hat_OH/K_D_FA);

  # Eq 7: Intermediate term for nu_FH_prime. The symbol H_hat_F is used.
  # Similar to H_hat_p, the paper's description of H_hat_F as 'free HIFa' is inconsistent with the formula. The model implements the formula as written.
  H_hat_F := 0.5 * (H_hat - F_hat_tot - K_hat_i_FH + sqrt((K_hat_i_FH + F_hat_tot - H_hat)^2 + 4 * K_hat_i_FH * H_hat));
  
  # Eq 9: FIH-dependent AR-hydroxylation
  # Note: The oxygen-dependent term was interpreted as a multiplier based on biochemical principles, correcting a likely typesetting error in the paper.
  nu_FA_prime := (k_cat_FA_prime * F_hat_tot) / (K_D_FA * (1 + H_hat_F/K_D_FH) + A_hat + gamma_ * A_hat_OH) * (O2_tilde / (alpha + O2_tilde));

  //// Rate Rules:
  # Eq 1: Rate of change for total HIFa (H_hat_tot)
  H_hat_tot' = 1 - H_hat_tot*(1 + nu_p_prime);
  
  # Eq 2: Rate of change for non-CAD-hydroxylated HIFa (H_hat)
  H_hat' = 1 - H_hat*(1 + nu_p_prime + nu_FH_prime);
  
  # Eq 3: Rate of change for unhydroxylated FIH target repeats (A_hat)
  A_hat' = (A_hat_tot - A_hat*(1 + eps*nu_FA_prime))/eps;

  //// Reactions:
  # All dynamics are defined by the ODEs in the Rate Rules section.

  //// Species initializations:
  # Initial conditions set to zero to simulate from a starting state.
  H_hat_tot = 0;
  H_hat = 0;
  A_hat = 0;

  //// Compartment initializations:
  cell = 1;

  //// Variable initializations:
  # Parameter values are from Table 2 of the paper.
  # Some parameters were varied in the paper; representative values are chosen here.
  P_hat_tot = 0.2;      # PHD concentration
  K_D_P = 1;            # Dissociation constant of PHD/HIFa binding
  k_cat_P_prime = 500;  # (kcat for ODD-hydroxylation)/(basal HIFa degradation rate constant)
  F_hat_tot = 1;        # FIH concentration
  K_D_FH = 1;           # Dissociation constant of FIH/HIFa binding
  
  # Parameter k_cat_FH_prime is inferred from text stating FIH turnover rate matches PHD's.
  k_cat_FH_prime = 500; # (kcat for HIFa CAD-hydroxylation)/(basal HIFa degradation rate constant)
  
  # Parameter k_cat_FA_prime is inferred from rho=1 (kcat_FH/kcat_FA), implying k_cat_FH_prime = k_cat_FA_prime
  k_cat_FA_prime = 500; # (kcat for ARD hydroxylation)/(basal HIFa degradation rate constant)
  
  alpha = 0.33;         # (KM of FIH for oxygen)/(KM of PHD for oxygen)
  A_hat_tot = 100;      # total concentration of FIH target ARs (varied in paper, 0-500)
  K_D_FA = 1;           # Dissociation constant of FIH/ARD protein binding (unhydroxylated ARs)
  gamma_ = 0.02;         # (FIH affinity for hydroxylated)/(FIH affinity for unhydroxylated ARs) (varied in paper, 0-0.1)
  eps = 5;              # (basal HIFa degradation rate constant)/(AR degradation rate constant) (varied in paper, 1-10)
  O2_tilde = 1;         # Oxygen concentration (varied in paper, 0-1)

  //// Other declarations:
  var H_hat_OH, A_hat_OH, nu_p_prime, H_hat_p, nu_FH_prime, K_hat_i_FH, H_hat_F, nu_FA_prime;
  const cell, P_hat_tot, K_D_P, k_cat_P_prime, F_hat_tot, K_D_FH, k_cat_FH_prime, k_cat_FA_prime;
  const alpha, A_hat_tot, K_D_FA, gamma_, eps, O2_tilde;

  //// Display Names:
  H_hat_tot is "Total HIF-alpha";
  H_hat is "Non-CAD-hydroxylated HIF-alpha";
  H_hat_OH is "CAD-hydroxylated HIF-alpha";
  A_hat is "Unhydroxylated FIH target repeats";
  A_hat_OH is "Hydroxylated FIH target repeats";
  A_hat_tot is "Total FIH target ankyrin repeats";
  P_hat_tot is "Total PHD";
  F_hat_tot is "Total FIH";
  O2_tilde is "Intracellular oxygen";
  
end
