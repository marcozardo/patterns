model *Kok2020_IFNalpha_Signaling()

  //// Compartments and Species:
  compartment cytoplasm, nucleus;

  species USP18mRNA in nucleus, SOCS1mRNA in nucleus, SOCS3mRNA in nucleus, IRF2mRNA in nucleus;
  species STAT1p in cytoplasm, STAT2p in cytoplasm, IRF9p in cytoplasm;
  species USP18p in cytoplasm, SOCS1p in cytoplasm, SOCS3p in cytoplasm, IRF2p in cytoplasm;

  //// Assignment Rules:
  total_STAT1 := STAT1c + STAT1n + pSTAT1c + pSTAT1n + 2*pSTAT1dimc + 2*pSTAT1dimn + pSTAT1pSTAT2c + pSTAT1pSTAT2n + ISGF3c + ISGF3n;
  total_STAT2 := STAT2c + STAT2n + pSTAT2c + pSTAT2n + pSTAT1pSTAT2c + pSTAT1pSTAT2n + ISGF3c + ISGF3n;
  total_IRF9 := IRF9c + IRF9n + ISGF3c + ISGF3n;

  //// Rate Rules:
  // (Note: These would require detailed ODE definitions from the original model)

  //// Reactions:
  // Receptor activation
  R1: IFN + Rec -> aRecIFN; k1*IFN*Rec;
  
  // STAT phosphorylation
  R2: aRecIFN + STAT1c -> aRecIFN + pSTAT1c; k2*aRecIFN*STAT1c;
  R3: aRecIFN + STAT2c -> aRecIFN + pSTAT2c; k3*aRecIFN*STAT2c;
  
  // Dimerization and complex formation
  R4: 2 pSTAT1c -> pSTAT1dimc; k4*pSTAT1c^2;
  R5: pSTAT1c + pSTAT2c -> pSTAT1pSTAT2c; k5*pSTAT1c*pSTAT2c;
  R6: pSTAT1pSTAT2c + IRF9c -> ISGF3c; k6*pSTAT1pSTAT2c*IRF9c;
  
  // Nuclear translocation
  R7: pSTAT1dimc -> pSTAT1dimn; k7*pSTAT1dimc;
  R8: pSTAT1pSTAT2c -> pSTAT1pSTAT2n; k8*pSTAT1pSTAT2c;
  R9: ISGF3c -> ISGF3n; k9*ISGF3c;
  
  // Transcription factor binding
  R10: pSTAT1dimn + OccGASbs -> OccGASbs; k10*pSTAT1dimn*OccGASbs;
  R11: pSTAT1pSTAT2n + OccGASbs + OccISREbs -> OccGASbs + OccISREbs; k11*pSTAT1pSTAT2n*(OccGASbs + OccISREbs);
  R12: ISGF3n + OccISREbs -> OccISREbs; k12*ISGF3n*OccISREbs;
  
  // Gene expression
  R13: OccGASbs -> STAT1mRNA; k13*OccGASbs;
  R14: OccGASbs + OccISREbs -> STAT2mRNA; k14*(OccGASbs + OccISREbs);
  R15: OccGASbs + OccISREbs -> IRF9mRNA; k15*(OccGASbs + OccISREbs);
  R16: OccGASbs + OccISREbs -> USP18mRNA; k16*(OccGASbs + OccISREbs);
  R17: OccGASbs + OccISREbs -> SOCS1mRNA; k17*(OccGASbs + OccISREbs);
  R18: OccGASbs -> SOCS3mRNA; k18*OccGASbs;
  R19: OccGASbs + OccISREbs -> IRF2mRNA; k19*(OccGASbs + OccISREbs);
  
  // Translation
  R20: STAT1mRNA -> STAT1p; k20*STAT1mRNA;
  R21: STAT2mRNA -> STAT2p; k21*STAT2mRNA;
  R22: IRF9mRNA -> IRF9p; k22*IRF9mRNA;
  R23: USP18mRNA -> USP18p; k23*USP18mRNA;
  R24: SOCS1mRNA -> SOCS1p; k24*SOCS1mRNA;
  R25: SOCS3mRNA -> SOCS3p; k25*SOCS3mRNA;
  R26: IRF2mRNA -> IRF2p; k26*IRF2mRNA;
  
  // Negative feedback mechanisms
  R27: USP18p + aRecIFN -> Rec; k27*USP18p*aRecIFN;
  R28: SOCS1p + aRecIFN -> Rec; k28*SOCS1p*aRecIFN;
  R29: SOCS3p + aRecIFN -> Rec; k29*SOCS3p*aRecIFN;
  R30: IRF2p + SOCS1mRNA -> SOCS1mRNA; k30*IRF2p*SOCS1mRNA;

  //// Events:
  E1: at time >= 0: IFN = IFN_prestim;
  E2: at time >= 1440: IFN = IFN_stim;  // 24 hours = 1440 minutes
