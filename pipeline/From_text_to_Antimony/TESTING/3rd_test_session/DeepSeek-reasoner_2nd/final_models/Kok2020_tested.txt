model *IFNalpha_Signal_Transduction()

  //// Compartments and Species:
  compartment cytoplasm, nucleus;
  species $IFNa in cytoplasm, Rec in cytoplasm, aRecIFN in cytoplasm;
  species STAT1c in cytoplasm, STAT2c in cytoplasm, IRF9c in cytoplasm;
  species pSTAT1c in cytoplasm, pSTAT2c in cytoplasm;
  species pSTAT1dimc in cytoplasm, pSTAT1pSTAT2c in cytoplasm, ISGF3c in cytoplasm;
  species STAT1n in nucleus, STAT2n in nucleus, IRF9n in nucleus;
  species pSTAT1n in nucleus, pSTAT2n in nucleus;
  species pSTAT1dimn in nucleus, pSTAT1pSTAT2n in nucleus, ISGF3n in nucleus;
  species OccGASbs in nucleus, OccISREbs in nucleus;
  species SOCS3mRNA in nucleus, USP18mRNA in nucleus, SOCS1mRNA in nucleus;
  species IRF2mRNA in nucleus, IRF9mRNA in nucleus, STAT1mRNA in nucleus, STAT2mRNA in nucleus;
  species SOCS3 in cytoplasm, USP18 in cytoplasm, SOCS1 in cytoplasm, IRF2 in cytoplasm;

  //// Assignment Rules:
  OccGASbs := pSTAT1dimn;
  OccISREbs := ISGF3n + pSTAT1pSTAT2n;
  total_STAT1 := STAT1c + STAT1n + pSTAT1c + pSTAT1n + 2*pSTAT1dimc + 2*pSTAT1dimn + pSTAT1pSTAT2c + pSTAT1pSTAT2n + ISGF3c + ISGF3n;
  total_STAT2 := STAT2c + STAT2n + pSTAT2c + pSTAT2n + pSTAT1pSTAT2c + pSTAT1pSTAT2n + ISGF3c + ISGF3n;
  total_IRF9 := IRF9c + IRF9n + ISGF3c + ISGF3n;

  //// Rate Rules:
  SOCS3mRNA' = k_syn_SOCS3 * OccGASbs - k_deg_SOCS3mRNA * SOCS3mRNA;
  USP18mRNA' = k_syn_USP18 * OccISREbs - k_deg_USP18mRNA * USP18mRNA;
  SOCS1mRNA' = k_syn_SOCS1 * OccISREbs - (k_deg_SOCS1mRNA + k_IRF2_enhance * IRF2) * SOCS1mRNA;
  IRF2mRNA' = k_syn_IRF2 * OccISREbs - k_deg_IRF2mRNA * IRF2mRNA;
  IRF9mRNA' = k_syn_IRF9 * OccISREbs - k_deg_IRF9mRNA * IRF9mRNA;
  STAT1mRNA' = k_syn_STAT1 * OccISREbs - k_deg_STAT1mRNA * STAT1mRNA;
  STAT2mRNA' = k_syn_STAT2 * OccISREbs - k_deg_STAT2mRNA * STAT2mRNA;
  SOCS3' = k_tl_SOCS3 * SOCS3mRNA - k_deg_SOCS3 * SOCS3;
  USP18' = k_tl_USP18 * USP18mRNA - k_deg_USP18 * USP18;
  SOCS1' = k_tl_SOCS1 * SOCS1mRNA - k_deg_SOCS1 * SOCS1;
  IRF2' = k_tl_IRF2 * IRF2mRNA - k_deg_IRF2 * IRF2;
  IRF9c' = k_tl_IRF9 * IRF9mRNA - k_deg_IRF9 * IRF9c + k_export_IRF9 * IRF9n - k_import_IRF9 * IRF9c;
  STAT1c' = k_tl_STAT1 * STAT1mRNA - k_deg_STAT1 * STAT1c + k_export_STAT1 * STAT1n - k_import_STAT1 * STAT1c;
  STAT2c' = k_tl_STAT2 * STAT2mRNA - k_deg_STAT2 * STAT2c + k_export_STAT2 * STAT2n - k_import_STAT2 * STAT2c;

  //// Reactions:
  R1: IFNa + Rec -> aRecIFN; k_bind_IFN * IFNa * Rec - k_unbind_IFN * aRecIFN;
  R2: aRecIFN -> Rec; k_deg_aRec * aRecIFN * (1 + k_SOCS1_deg * SOCS1 + k_USP18_SOCS1_deg * SOCS1 * USP18);
  R3: STAT1c -> pSTAT1c; k_phos_STAT1 * aRecIFN * STAT1c / (1 + k_SOCS3_inh * SOCS3);
  R4: STAT2c -> pSTAT2c; k_phos_STAT2 * aRecIFN * STAT2c / (1 + k_SOCS3_inh * SOCS3);
  R5: pSTAT1c -> STAT1c; k_dephos_STAT1 * pSTAT1c;
  R6: pSTAT2c -> STAT2c; k_dephos_STAT2 * pSTAT2c;
  R7: 2 pSTAT1c -> pSTAT1dimc; k_dim_STAT1 * pSTAT1c^2;
  R8: pSTAT1dimc -> 2 pSTAT1c; k_diss_STAT1 * pSTAT1dimc;
  R9: pSTAT1c + pSTAT2c -> pSTAT1pSTAT2c; k_het_dim * pSTAT1c * pSTAT2c;
  R10: pSTAT1pSTAT2c -> pSTAT1c + pSTAT2c; k_het_diss * pSTAT1pSTAT2c;
  R11: pSTAT1pSTAT2c + IRF9c -> ISGF3c; k_ISGF3_form * pSTAT1pSTAT2c * IRF9c;
  R12: ISGF3c -> pSTAT1pSTAT2c + IRF9c; k_ISGF3_diss * ISGF3c;
  R13: pSTAT1dimc -> pSTAT1dimn; k_import_pSTAT1dim * pSTAT1dimc;
  R14: pSTAT1pSTAT2c -> pSTAT1pSTAT2n; k_import_pSTAT1pSTAT2 * pSTAT1pSTAT2c;
  R15: ISGF3c -> ISGF3n; k_import_ISGF3 * ISGF3c;
  R16: pSTAT1dimn -> pSTAT1dimc; k_export_pSTAT1dim * pSTAT1dimn;
  R17: pSTAT1pSTAT2n -> pSTAT1pSTAT2c; k_export_pSTAT1pSTAT2 * pSTAT1pSTAT2n;
  R18: ISGF3n -> ISGF3c; k_export_ISGF3 * ISGF3n;
  R19: STAT1n -> STAT1c; k_export_STAT1 * STAT1n;
  R20: STAT2n -> STAT2c; k_export_STAT2 * STAT2n;
  R21: IRF9n -> IRF9c; k_export_IRF9 * IRF9n;
  R22: STAT1c -> STAT1n; k_import_STAT1 * STAT1c;
  R23: STAT2c -> STAT2n; k_import_STAT2 * STAT2c;
  R24: IRF9c -> IRF9n; k_import_IRF9 * IRF9c;

  //// Species initializations:
  IFNa = 0;
  Rec = 100000;
  aRecIFN = 0;
  STAT1c = 500000;
  STAT2c = 50000;
  IRF9c = 5000;
  pSTAT1c = 0;
  pSTAT2c = 0;
  pSTAT1dimc = 0;
  pSTAT1pSTAT2c = 0;
  ISGF3c = 0;
  STAT1n = 0;
  STAT2n = 0;
  IRF9n = 0;
  pSTAT1n = 0;
  pSTAT2n = 0;
  pSTAT1dimn = 0;
  pSTAT1pSTAT2n = 0;
  ISGF3n = 0;
  OccGASbs = 0;
  OccISREbs = 0;
  SOCS3mRNA = 0;
  USP18mRNA = 0;
  SOCS1mRNA = 0;
  IRF2mRNA = 0;
  IRF9mRNA = 0;
  STAT1mRNA = 0;
  STAT2mRNA = 0;
  SOCS3 = 0;
  USP18 = 0;
  SOCS1 = 0;
  IRF2 = 0;

  //// Compartment initializations:
  cytoplasm = 1;
  nucleus = 1;

  //// Variable initializations:
  k_bind_IFN = 1e-6;
  k_unbind_IFN = 0.1;
  k_deg_aRec = 0.01;
  k_SOCS1_deg = 1e-4;
  k_USP18_SOCS1_deg = 1e-5;
  k_phos_STAT1 = 1e-5;
  k_phos_STAT2 = 1e-5;
  k_SOCS3_inh = 1e-4;
  k_dephos_STAT1 = 0.1;
  k_dephos_STAT2 = 0.1;
  k_dim_STAT1 = 1e-6;
  k_diss_STAT1 = 0.01;
  k_het_dim = 1e-6;
  k_het_diss = 0.01;
  k_ISGF3_form = 1e-6;
  k_ISGF3_diss = 0.01;
  k_import_pSTAT1dim = 0.1;
  k_import_pSTAT1pSTAT2 = 0.1;
  k_import_ISGF3 = 0.1;
  k_export_pSTAT1dim = 0.01;
  k_export_pSTAT1pSTAT2 = 0.01;
  k_export_ISGF3 = 0.01;
  k_export_STAT1 = 0.01;
  k_export_STAT2 = 0.01;
  k_export_IRF9 = 0.01;
  k_import_STAT1 = 0.1;
  k_import_STAT2 = 0.1;
  k_import_IRF9 = 0.1;
  k_syn_SOCS3 = 0.1;
  k_deg_SOCS3mRNA = 0.1;

  k_deg_USP18mRNA = 0.05;
  k_syn_SOCS1 = 0.05;

  k_IRF2_enhance = 1e-5;
  k_syn_IRF2 = 0.05;
  k_deg_IRF2mRNA = 0.05;
  k_syn_IRF9 = 0.1;
  k_deg_IRF9mRNA = 0.05;
  k_syn_STAT1 = 0.05;
  k_deg_STAT1mRNA = 0.05;
  k_syn_STAT2 = 0.05;
  k_deg_STAT2mRNA = 0.05;
  k_tl_SOCS3 = 0.1;
  k_deg_SOCS3 = 0.1;
  k_tl_USP18 = 0.05;
  k_deg_USP18 = 0.05;
  k_tl_SOCS1 = 0.05;
  k_deg_SOCS1 = 0.05;
  k_tl_IRF2 = 0.05;
  k_deg_IRF2 = 0.05;
  k_tl_IRF9 = 0.1;
  k_deg_IRF9 = 0.05;
  k_tl_STAT1 = 0.05;
  k_deg_STAT1 = 0.05;
  k_tl_STAT2 = 0.05;
  k_deg_STAT2 = 0.05;

  //// Other declarations:
  var OccGASbs, OccISREbs, total_STAT1, total_STAT2, total_IRF9;
  var SOCS3mRNA, USP18mRNA, SOCS1mRNA, IRF2mRNA, IRF9mRNA, STAT1mRNA, STAT2mRNA;
  var SOCS3, USP18, SOCS1, IRF2, IRF9c, STAT1c, STAT2c;
  const cytoplasm, nucleus, k_bind_IFN, k_unbind_IFN, k_deg_aRec, k_SOCS1_deg, k_USP18_SOCS1_deg, k_phos_STAT1, k_phos_STAT2, k_SOCS3_inh, k_dephos_STAT1, k_dephos_STAT2, k_dim_STAT1, k_diss_STAT1, k_het_dim, k_het_diss, k_ISGF3_form, k_ISGF3_diss, k_import_pSTAT1dim, k_import_pSTAT1pSTAT2, k_import_ISGF3, k_export_pSTAT1dim, k_export_pSTAT1pSTAT2, k_export_ISGF3, k_export_STAT1, k_export_STAT2, k_export_IRF9, k_import_STAT1, k_import_STAT2, k_import_IRF9, k_syn_SOCS3, k_deg_SOCS3mRNA, k_syn_USP18, k_deg_USP18mRNA, k_syn_SOCS1, k_deg_SOCS1mRNA, k_IRF2_enhance, k_syn_IRF2, k_deg_IRF2mRNA, k_syn_IRF9, k_deg_IRF9mRNA, k_syn_STAT1, k_deg_STAT1mRNA, k_syn_STAT2, k_deg_STAT2mRNA, k_tl_SOCS3, k_deg_SOCS3, k_tl_USP18, k_deg_USP18, k_tl_SOCS1, k_deg_SOCS1, k_tl_IRF2, k_deg_IRF2, k_tl_IRF9, k_deg_IRF9, k_tl_STAT1, k_deg_STAT1, k_tl_STAT2, k_deg_STAT2;

  //// Unit definitions:
  unit substance = 1e-9 mole;
  unit time_unit = 1 second;

  IFNa is "Interferon-alpha";
