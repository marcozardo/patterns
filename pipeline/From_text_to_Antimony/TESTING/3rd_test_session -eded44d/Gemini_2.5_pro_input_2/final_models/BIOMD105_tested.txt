  J21: DUB + MisP_Ub2 => DUB + MisP_Ub + Ub; k66*DUB*MisP_Ub2;
  J22: DUB + MisP_Ub => DUB + MisP + Ub; k66*DUB*MisP_Ub;

  J24: MisP_Ub5 + Proteasome => MisP_Ub5_Proteasome; k67*MisP_Ub5*Proteasome;
  J25: MisP_Ub6 + Proteasome => MisP_Ub6_Proteasome; k67*MisP_Ub6*Proteasome;

  J27: MisP_Ub8 + Proteasome => MisP_Ub8_Proteasome; k67*MisP_Ub8*Proteasome;

  J32: DUB + MisP_Ub4_Proteasome => DUB + MisP_Ub3 + Proteasome + Ub; k68*DUB*MisP_Ub4_Proteasome; # Release from proteasome

  # Degradation by Proteasome
