  //// Reactions:
  # Receptor dynamics: activation and negative feedback
  J1: IFN + Rec => aRecIFN; k_Rec_act * IFN * Rec;
  J2: aRecIFN => Rec; k_Rec_deact * aRecIFN;
  J3: aRecIFN + USP18_c => Rec + USP18_c; k_inh_USP18 * aRecIFN * USP18_c;
  J4: aRecIFN + SOCS3_c => Rec + SOCS3_c; k_inh_SOCS3 * aRecIFN * SOCS3_c;
  J5: aRecIFN + SOCS1_c => SOCS1_c; k_degrad_SOCS1 * aRecIFN * SOCS1_c;
  J6: aRecIFN + SOCS1_c + USP18_c => SOCS1_c + USP18_c; k_degrad_SOCS1_USP18 * aRecIFN * SOCS1_c * USP18_c;

  # Cytoplasmic signaling: STAT phosphorylation and complex formation
  J7: 2 STAT1_c => pSTAT1_dim_c; k_pSTAT1_dim * aRecIFN * STAT1_c^2 / (Km_pSTAT1 + STAT1_c^2);
  J8: STAT1_c + STAT2_c => pSTAT1_pSTAT2_c; k_pSTAT1_pSTAT2 * aRecIFN * STAT1_c * STAT2_c / ((Km_pSTAT1_pSTAT2_S1 + STAT1_c) * (Km_pSTAT1_pSTAT2_S2 + STAT2_c));
  J9: pSTAT1_pSTAT2_c + IRF9_c => ISGF3_c; k_ISGF3_form * pSTAT1_pSTAT2_c * IRF9_c;
  J10: ISGF3_c => pSTAT1_pSTAT2_c + IRF9_c; k_ISGF3_diss * ISGF3_c;

  # Nuclear transport
  J11: pSTAT1_dim_c => pSTAT1_dim_n; k_import_pSTAT1_dim * pSTAT1_dim_c;
  J12: pSTAT1_pSTAT2_c => pSTAT1_pSTAT2_n; k_import_pSTAT1_pSTAT2 * pSTAT1_pSTAT2_c;
  J13: ISGF3_c => ISGF3_n; k_import_ISGF3 * ISGF3_c;
  J14: STAT1_c -> STAT1_n; k_import_STAT1 * STAT1_c;
  J15: STAT1_n -> STAT1_c; k_export_STAT1 * STAT1_n;
  J16: STAT2_c -> STAT2_n; k_import_STAT2 * STAT2_c;
  J17: STAT2_n -> STAT2_c; k_export_STAT2 * STAT2_n;
  J18: IRF9_c -> IRF9_n; k_import_IRF9 * IRF9_c;
  J19: IRF9_n -> IRF9_c; k_export_IRF9 * IRF9_n;

  # Nuclear complex dissociation and dephosphorylation
  J20: pSTAT1_dim_n => 2 STAT1_n; k_diss_pSTAT1_dim * pSTAT1_dim_n;
  J21: pSTAT1_pSTAT2_n => STAT1_n + STAT2_n; k_diss_pSTAT1_pSTAT2 * pSTAT1_pSTAT2_n;
  J22: ISGF3_n => pSTAT1_pSTAT2_n + IRF9_n; k_diss_ISGF3 * ISGF3_n;

  # --- Transcriptional Feedback Loops ---
  # SOCS3 gene expression (GAS-driven)
  J23: => SOCS3_mRNA; Vm_SOCS3 * pSTAT1_dim_n / (Km_SOCS3 + pSTAT1_dim_n);
  J24: SOCS3_mRNA => ; k_deg_SOCS3_mRNA * SOCS3_mRNA;
  J25: SOCS3_mRNA => SOCS3_mRNA + SOCS3_c; k_tl_SOCS3 * SOCS3_mRNA;
  J26: SOCS3_c => ; k_deg_SOCS3 * SOCS3_c;

  # STAT1 gene expression (GAS/ISRE-driven)
  J27: => STAT1_mRNA; V_STAT1_base + (V_STAT1_S1S2 * pSTAT1_pSTAT2_n / (K_STAT1_S1S2 + pSTAT1_pSTAT2_n) + V_STAT1_ISGF3 * ISGF3_n / (K_STAT1_ISGF3 + ISGF3_n));
  J28: STAT1_mRNA => ; k_deg_STAT1_mRNA * STAT1_mRNA;
  J29: STAT1_mRNA => STAT1_mRNA + STAT1_c; k_tl_STAT1 * STAT1_mRNA;
  J30: STAT1_c => ; k_deg_STAT1_c * STAT1_c;
  J31: STAT1_n => ; k_deg_STAT1_n * STAT1_n;

  # STAT2 gene expression (GAS/ISRE-driven)
  J32: => STAT2_mRNA; V_STAT2_base + (V_STAT2_S1S2 * pSTAT1_pSTAT2_n / (K_STAT2_S1S2 + pSTAT1_pSTAT2_n) + V_STAT2_ISGF3 * ISGF3_n / (K_STAT2_ISGF3 + ISGF3_n));
  J33: STAT2_mRNA => ; k_deg_STAT2_mRNA * STAT2_mRNA;
  J34: STAT2_mRNA => STAT2_mRNA + STAT2_c; k_tl_STAT2 * STAT2_mRNA;
  J35: STAT2_c => ; k_deg_STAT2_c * STAT2_c;
  J36: STAT2_n => ; k_deg_STAT2_n * STAT2_n;
  
  # IRF9 gene expression (GAS/ISRE-driven)
  J37: => IRF9_mRNA; V_IRF9_base + (V_IRF9_S1S2 * pSTAT1_pSTAT2_n / (K_IRF9_S1S2 + pSTAT1_pSTAT2_n) + V_IRF9_ISGF3 * ISGF3_n / (K_IRF9_ISGF3 + ISGF3_n));
  J38: IRF9_mRNA => ; k_deg_IRF9_mRNA * IRF9_mRNA;
  J39: IRF9_mRNA => IRF9_mRNA + IRF9_c; k_tl_IRF9 * IRF9_mRNA;
  J40: IRF9_c => ; k_deg_IRF9_c * IRF9_c;
  J41: IRF9_n => ; k_deg_IRF9_n * IRF9_n;

  # USP18 gene expression (GAS/ISRE-driven)

  J43: USP18_mRNA => ; k_deg_USP18_mRNA * USP18_mRNA;
  J44: USP18_mRNA => USP18_mRNA + USP18_c; k_tl_USP18 * USP18_mRNA;
  J45: USP18_c => ; k_deg_USP18 * USP18_c;

  # SOCS1 gene expression (GAS/ISRE-driven, with IRF2 regulation)
  J46: => SOCS1_mRNA; V_SOCS1_base + (V_SOCS1_S1S2 * pSTAT1_pSTAT2_n / (K_SOCS1_S1S2 + pSTAT1_pSTAT2_n) + V_SOCS1_ISGF3 * ISGF3_n / (K_SOCS1_ISGF3 + ISGF3_n));
  J47: SOCS1_mRNA => ; (k_deg_SOCS1_mRNA + k_deg_SOCS1_IRF2 * IRF2_c) * SOCS1_mRNA;
  J48: SOCS1_mRNA => SOCS1_mRNA + SOCS1_c; k_tl_SOCS1 * SOCS1_mRNA;
  J49: SOCS1_c => ; k_deg_SOCS1 * SOCS1_c;
  
  # IRF2 gene expression (GAS/ISRE-driven)
  J50: => IRF2_mRNA; V_IRF2_base + (V_IRF2_S1S2 * pSTAT1_pSTAT2_n / (K_IRF2_S1S2 + pSTAT1_pSTAT2_n) + V_IRF2_ISGF3 * ISGF3_n / (K_IRF2_ISGF3 + ISGF3_n));
  J51: IRF2_mRNA => ; k_deg_IRF2_mRNA * IRF2_mRNA;
  J52: IRF2_mRNA => IRF2_mRNA + IRF2_c; k_tl_IRF2 * IRF2_mRNA;
  J53: IRF2_c => ; k_deg_IRF2 * IRF2_c;

  E_stim: at (time >= t_stim): IFN = IFN_stim;
