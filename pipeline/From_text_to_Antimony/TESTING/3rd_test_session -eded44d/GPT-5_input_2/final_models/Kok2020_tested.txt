  species $IFN in Cyt;
  species Rec in Cyt, aRecIFN in Cyt, iRec in Cyt, iRecS1 in Cyt;

  species STAT1c in Cyt, STAT2c in Cyt, IRF9c in Cyt;
  species STAT1n in Nuc, STAT2n in Nuc, IRF9n in Nuc;

  species pSTAT1dimc in Cyt, pSTAT1pSTAT2c in Cyt, ISGF3c in Cyt;
  species pSTAT1dimn in Nuc, pSTAT1pSTAT2n in Nuc, ISGF3n in Nuc;

  species SOCS1 in Cyt, SOCS3 in Cyt, USP18 in Cyt, IRF2 in Cyt;

  species STAT1mRNA in Cyt, STAT2mRNA in Cyt, IRF9mRNA in Cyt, USP18mRNA in Cyt, SOCS1mRNA in Cyt, SOCS3mRNA in Cyt, IRF2mRNA in Cyt;

  species OccGASbs in Nuc, OccISREbs in Nuc, OccGASISREbs in Nuc;

  //// Assignment Rules:
  // none

  //// Rate Rules:
  // none

  //// Reactions:
  # IFN binding and receptor activation/inactivation
  R1_bind: $IFN + Rec => aRecIFN; k_on_IFNRec*$IFN*Rec;
  R2_unbind: aRecIFN => Rec; k_off_IFNRec*aRecIFN;

  # Receptor inactivation/turnover (basal and feedback-mediated)
  R3_deg_bas: aRecIFN => ; k_deg_aRec_bas*aRecIFN;
  R4_deg_SOCS1: aRecIFN => ; k_deg_aRec_SOCS1*SOCS1*aRecIFN;
  R5_deg_SOCS1_USP18: aRecIFN => ; k_deg_aRec_SOCS1_USP18*SOCS1*USP18*aRecIFN;
  R6_deg_SOCS3: aRecIFN => ; k_deg_aRec_SOCS3*SOCS3*aRecIFN;

  # USP18 and SOCS1 reversible inhibitory complex formation with active receptor
  R7_bind_USP18: aRecIFN + USP18 => iRec; k_bind_USP18*aRecIFN*USP18;
  R8_unbind_USP18: iRec => Rec + USP18; k_unbind_USP18*iRec;

  R9_bind_SOCS1: aRecIFN + SOCS1 => iRecS1; k_bind_SOCS1*aRecIFN*SOCS1;
  R10_unbind_SOCS1: iRecS1 => Rec + SOCS1; k_unbind_SOCS1*iRecS1;

  # STAT phosphorylation/dimerization catalyzed by active receptor (approximated)
  R11_pSTAT1dim_form: 2 STAT1c => pSTAT1dimc; k_pSTAT1dim*aRecIFN*STAT1c^2;
  R12_pSTAT1pSTAT2_form: STAT1c + STAT2c => pSTAT1pSTAT2c; k_pSTAT1pSTAT2*aRecIFN*STAT1c*STAT2c;

  # Cytoplasmic complex dissociation
  R13_pSTAT1dimc_diss: pSTAT1dimc => 2 STAT1c; k_diss_pSTAT1dimc*pSTAT1dimc;
  R14_pSTAT1pSTAT2c_diss: pSTAT1pSTAT2c => STAT1c + STAT2c; k_diss_pSTAT1pSTAT2c*pSTAT1pSTAT2c;

  # ISGF3 cytoplasmic formation/dissociation
  R15_ISGF3c_form: pSTAT1pSTAT2c + IRF9c => ISGF3c; k_form_ISGF3c*pSTAT1pSTAT2c*IRF9c;
  R16_ISGF3c_diss: ISGF3c => pSTAT1pSTAT2c + IRF9c; k_diss_ISGF3c*ISGF3c;

  # Nuclear translocation of complexes
  R17_imp_pSTAT1dim: pSTAT1dimc => pSTAT1dimn; k_imp_pSTAT1dim*pSTAT1dimc;
  R18_exp_pSTAT1dim: pSTAT1dimn => pSTAT1dimc; k_exp_pSTAT1dim*pSTAT1dimn;

  R19_imp_pSTAT1pSTAT2: pSTAT1pSTAT2c => pSTAT1pSTAT2n; k_imp_pSTAT1pSTAT2*pSTAT1pSTAT2c;
  R20_exp_pSTAT1pSTAT2: pSTAT1pSTAT2n => pSTAT1pSTAT2c; k_exp_pSTAT1pSTAT2*pSTAT1pSTAT2n;

  R21_imp_ISGF3: ISGF3c => ISGF3n; k_imp_ISGF3*ISGF3c;
  R22_exp_ISGF3: ISGF3n => ISGF3c; k_exp_ISGF3*ISGF3n;

  # Monomer shuttling
  R23_imp_STAT1: STAT1c => STAT1n; k_imp_STAT1*STAT1c;
  R24_exp_STAT1: STAT1n => STAT1c; k_exp_STAT1*STAT1n;

  R25_imp_STAT2: STAT2c => STAT2n; k_imp_STAT2*STAT2c;
  R26_exp_STAT2: STAT2n => STAT2c; k_exp_STAT2*STAT2n;

  R27_imp_IRF9: IRF9c => IRF9n; k_imp_IRF9*IRF9c;
  R28_exp_IRF9: IRF9n => IRF9c; k_exp_IRF9*IRF9n;

  # Nuclear complex formation/dissociation
  R29_ISGF3n_form: pSTAT1pSTAT2n + IRF9n => ISGF3n; k_form_ISGF3n*pSTAT1pSTAT2n*IRF9n;
  R30_ISGF3n_diss: ISGF3n => pSTAT1pSTAT2n + IRF9n; k_diss_ISGF3n*ISGF3n;

  R31_pSTAT1dimn_diss: pSTAT1dimn => 2 STAT1n; k_diss_pSTAT1dimn*pSTAT1dimn;
  R32_pSTAT1pSTAT2n_diss: pSTAT1pSTAT2n => STAT1n + STAT2n; k_diss_pSTAT1pSTAT2n*pSTAT1pSTAT2n;

  # Promoter occupancy dynamics (effective)
  R33_occGAS_prod:  => OccGASbs; k_on_GAS*pSTAT1dimn;
  R34_occGAS_loss: OccGASbs => ; k_off_GAS*OccGASbs;

  R35_occISRE_prod:  => OccISREbs; k_on_ISRE*ISGF3n;
  R36_occISRE_loss: OccISREbs => ; k_off_ISRE*OccISREbs;

  R37_occGASISRE_prod:  => OccGASISREbs; k_on_GASISRE_ISGF3*ISGF3n + k_on_GASISRE_HET*pSTAT1pSTAT2n;
  R38_occGASISRE_loss: OccGASISREbs => ; k_off_GASISRE*OccGASISREbs;

  # Transcription (mRNA synthesis) and degradation
  R39_tx_STAT1:  => STAT1mRNA; k_tx_STAT1*OccGASISREbs + k_basal_m_STAT1;
  R40_deg_STAT1m: STAT1mRNA => ; k_deg_m_STAT1*STAT1mRNA;

  R41_tx_STAT2:  => STAT2mRNA; k_tx_STAT2*OccGASISREbs + k_basal_m_STAT2;
  R42_deg_STAT2m: STAT2mRNA => ; k_deg_m_STAT2*STAT2mRNA;

  R43_tx_IRF9:  => IRF9mRNA; k_tx_IRF9*OccGASISREbs + k_basal_m_IRF9;
  R44_deg_IRF9m: IRF9mRNA => ; k_deg_m_IRF9*IRF9mRNA;

  R45_tx_USP18:  => USP18mRNA; k_tx_USP18*OccGASISREbs + k_basal_m_USP18;
  R46_deg_USP18m: USP18mRNA => ; k_deg_m_USP18*USP18mRNA;

  R47_tx_SOCS1:  => SOCS1mRNA; k_tx_SOCS1*OccGASISREbs + k_basal_m_SOCS1;
  R48_deg_SOCS1m: SOCS1mRNA => ; k_deg_m_SOCS1*SOCS1mRNA;
  R49_deg_SOCS1m_IRF2: SOCS1mRNA + IRF2 => ; k_deg_m_SOCS1_IRF2*SOCS1mRNA*IRF2;

  R50_tx_SOCS3:  => SOCS3mRNA; k_tx_SOCS3*OccGASbs + k_basal_m_SOCS3;
  R51_deg_SOCS3m: SOCS3mRNA => ; k_deg_m_SOCS3*SOCS3mRNA;

  R52_tx_IRF2:  => IRF2mRNA; k_tx_IRF2*OccGASISREbs + k_basal_m_IRF2;
  R53_deg_IRF2m: IRF2mRNA => ; k_deg_m_IRF2*IRF2mRNA;

  # Translation (protein synthesis) and degradation
  R54_tl_STAT1: STAT1mRNA => STAT1c; k_tl_STAT1*STAT1mRNA;
  R55_deg_STAT1c: STAT1c => ; k_deg_p_STAT1c*STAT1c;
  R56_deg_STAT1n: STAT1n => ; k_deg_p_STAT1n*STAT1n;

  R57_tl_STAT2: STAT2mRNA => STAT2c; k_tl_STAT2*STAT2mRNA;
  R58_deg_STAT2c: STAT2c => ; k_deg_p_STAT2c*STAT2c;
  R59_deg_STAT2n: STAT2n => ; k_deg_p_STAT2n*STAT2n;

  R60_tl_IRF9: IRF9mRNA => IRF9c; k_tl_IRF9*IRF9mRNA;
  R61_deg_IRF9c: IRF9c => ; k_deg_p_IRF9c*IRF9c;
  R62_deg_IRF9n: IRF9n => ; k_deg_p_IRF9n*IRF9n;

  R63_tl_USP18: USP18mRNA => USP18; k_tl_USP18*USP18mRNA;
  R64_deg_USP18: USP18 => ; k_deg_p_USP18*USP18;

  R65_tl_SOCS1: SOCS1mRNA => SOCS1; k_tl_SOCS1*SOCS1mRNA;
  R66_deg_SOCS1: SOCS1 => ; k_deg_p_SOCS1*SOCS1;

  R67_tl_SOCS3: SOCS3mRNA => SOCS3; k_tl_SOCS3*SOCS3mRNA;
  R68_deg_SOCS3: SOCS3 => ; k_deg_p_SOCS3*SOCS3;

  R69_tl_IRF2: IRF2mRNA => IRF2; k_tl_IRF2*IRF2mRNA;
  R70_deg_IRF2: IRF2 => ; k_deg_p_IRF2*IRF2;

/*
Unmapped or non-numerical details from the source (not explicitly parameterized here):
- Linear-chain delays between mRNA and protein translation were described but no chain lengths/rates provided; here implemented as single-step translation.
- Distinct relative strengths of ISGF3 vs pSTAT1:pSTAT2 in driving GAS+ISRE sites are represented via k_on_GASISRE_ISGF3 > k_on_GASISRE_HET but exact values were not provided.
- SOCS1-mediated and SOCS1:USP18-mediated degradation of active receptor is represented via additive catalytic degradation terms; alternative structural variants tested in the paper are not distinguished further.
- Additional cytoplasmic phosphatase activity dissociating complexes was rejected in the paper; only simple dissociation terms are included here.
- Exact parameter values (85+ in the original model) and initial conditions were calibrated in the publication; placeholder values are used here for executability and should be re-estimated for quantitative studies.
