model *Queralt2006_MitoticExit_Core()

  //// Compartments and Species:
  compartment cell;
  species Cdc20 in cell, Securin in cell, PP2A in cell, Net1P in cell, Cdc14 in cell, MEN in cell, Cdh1 in cell, Clb2 in cell;

  //// Assignment Rules:
  Cdk := Clb2;                                  // Clb2-Cdk activity proxy
  Sepa := Sep_tot/(1 + (Securin/Ki_sec));       // separase activity increases as securin declines (inhibitory Hill-like form)

  //// Rate Rules:
  Cdc20' = k_syn20 - k_deg20*Cdc20;                                                                                              // Cdc20 synthesis is induced at t=0 (metaphase release), then turns over
  Securin' = k_synSec - k_apc20*Cdc20*Securin - k_bdeg_sec*Securin;                                                                // APC^Cdc20-dependent securin destruction
  PP2A' = k_Prec*(PP2A0 - PP2A) - k_Pdown*Sepa*PP2A;                                                                              // separase-dependent downregulation of PP2A^Cdc55 at anaphase onset
  Net1P' = k_cdk_net1*Cdk*(1 - Net1P) + k_men_net1*MEN*(1 - Net1P) - k_pp2a_net1*PP2A*Net1P - k_cdc14_net1*Cdc14*Net1P;          // Net1 phosphorylation by Cdk and MEN; dephosphorylation by PP2A^Cdc55 and Cdc14
  Cdc14' = k_release*(Net1P + k_men_c14*MEN)*(1 - Cdc14) - k_sequestration*(1 - Net1P)*Cdc14;                                     // Cdc14 release upon Net1 phosphorylation; resequestration when Net1 is unphosphorylated; MEN sustains release
  MEN' = k_act_men*(Cdc14/(1 + k_inh_cdk*Clb2))*(1 - MEN) + k_bfa1*(1 - PP2A)*(1 - MEN) - k_inact_men*MEN;                        // MEN activated by Cdc14 and by Bfa1-P (favored when PP2A is low); inhibited by high Cdk; inactivates otherwise
  Cdh1' = k_act_cdh1*Cdc14*(1 - Cdh1) - k_inh_cdh1*Clb2*Cdh1;                                                                      // Cdc14 activates Cdh1; Cdk inhibits Cdh1
  Clb2' = k_syn_clb2 - (k_apc20_cyc*Cdc20 + k_apc_cdh1*Cdh1)*Clb2;                                                                 // Cyclin destruction by APC^Cdc20 early and APC^Cdh1 later
