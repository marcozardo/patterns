  compartment cell;

  k11_T := k10*(Vden/(1 - Vden));

  l7_T := piecewise(l7_37, T <= 37, (l7_41 - l7_37)/(41 - 37)*(T - 37) + l7_37, (T > 37) && (T <= 41), (l7_42 - l7_41)/(42 - 41)*(T - 41) + l7_41, (T > 41) && (T <= 42), (l7_43 - l7_42)/(43 - 42)*(T - 42) + l7_42, (T > 42) && (T <= 43), (l7_44 - l7_43)/(44 - 43)*(T - 43) + l7_43, (T > 43) && (T <= 44), (l7_45 - l7_44)/(45 - 44)*(T - 44) + l7_44, (T > 44) && (T <= 45));

  # Assuming the text description is more complete.

  # thus the reverse reaction of P<->S is implicitly handled by the protein refolding pathway.
  
  # translation
