model *Queralt2006_MitoticExit_BuddingYeast()

  // Compartments and Species:
  compartment cell;
  species Cdc14 in cell, Net1 in cell, PP2A_Cdc55 in cell, Separase in cell, Securin in cell;
  species Cdk_Clb2 in cell, Cdc20 in cell, Cdh1 in cell, Polo in cell, Bfa1 in cell, Bub2 in cell;
  species MEN in cell, Sic1 in cell;
  species Cdc14_Net1_complex in cell, Net1_P in cell, PP2A_Cdc55_inactive in cell;

  // Assignment Rules:
  Active_Cdc14 := Cdc14 * (1 - Net1_inhibition);
  Net1_inhibition := 1/(1 + exp(-k_Net1*(Net1_P - Net1_P_threshold)));
  MEN_activity := (Active_Cdc14 * k1_MEN * (1 - Bfa1_inhibition)) / (k2_MEN + (1 - Bfa1_inhibition));
  Bfa1_inhibition := 1/(1 + exp(-k_Bfa1*(Bfa1_P - Bfa1_P_threshold)));
  Bfa1_P := (Polo * k1_Bfa1_phos * Bfa1) / (k2_Bfa1_phos + Bfa1) - (PP2A_Cdc55 * k3_Bfa1_dephos * Bfa1_P) / (k4_Bfa1_dephos + Bfa1_P);

  // Rate Rules:
  Cdc14' = -k_seq * Cdc14 * Net1 + k_rel * Net1_P * Cdc14_Net1_complex - k_deact * Active_Cdc14;
  Net1' = -k_phos * Cdk_Clb2 * Net1 + k_dephos * PP2A_Cdc55 * Net1_P;
  PP2A_Cdc55' = -k_downreg * Separase * PP2A_Cdc55 + k_react * PP2A_Cdc55_inactive;
  Separase' = k_act * (Securin_degradation) - k_inact * Separase;
  Cdk_Clb2' = -k_deg * Cdc20 * Cdk_Clb2 - k_deg2 * Cdh1 * Cdk_Clb2;
  Net1_P' = k_phos * Cdk_Clb2 * Net1 - k_dephos * PP2A_Cdc55 * Net1_P;

  // Reactions:
  Securin_degradation: Securin -> ; k_secdeg * Cdc20 * Securin;
  Cdc14_release: Cdc14_Net1_complex -> Cdc14 + Net1_P; k_rel * Net1_P * Cdc14_Net1_complex;
  Net1_phosphorylation: Net1 -> Net1_P; k_phos * Cdk_Clb2 * Net1;
  Net1_dephosphorylation: Net1_P -> Net1; k_dephos * PP2A_Cdc55 * Net1_P;
  PP2A_downregulation: PP2A_Cdc55 -> PP2A_Cdc55_inactive; k_downreg * Separase * PP2A_Cdc55;
  Cdk_inactivation: Cdk_Clb2 -> ; k_deg * Cdc20 * Cdk_Clb2 + k_deg2 * Cdh1 * Cdk_Clb2;
  MEN_activation: -> MEN; k_MEN_act * Active_Cdc14 * (1 - Bfa1_inhibition);

  // Events:
  at (time >= anaphase_onset): Separase = Separase_0;
  at (Cdk_Clb2 < Cdk_threshold): Cdh1 = Cdh1_0;

  // Species initializations:
  Cdc14 = 1.0;
  Net1 = 1.0;
  PP2A_Cdc55 = 1.0;
  Separase = 0.0;
  Securin = 1.0;
  Cdk_Clb2 = 1.0;
  Cdc20 = 0.0;
  Cdh1 = 0.0;
  Polo = 0.5;
  Bfa1 = 1.0;
  Bub2 = 1.0;
  MEN = 0.0;
  Sic1 = 0.0;
  Cdc14_Net1_complex = 1.0;
  Net1_P = 0.0;
  PP2A_Cdc55_inactive = 0.0;

  // Compartment initializations:
  cell = 1.0;

  // Variable initializations:
  k_seq = 1.0;
  k_rel = 1.0;
  k_deact = 0.1;
  k_phos = 1.0;
  k_dephos = 1.0;
  k_downreg = 1.0;
  k_react = 0.1;
  k_act = 1.0;
  k_inact = 0.1;
  k_secdeg = 1.0;
  k_deg = 0.1;
  k_deg2 = 0.1;
  k_Net1 = 10.0;
  Net1_P_threshold = 0.5;
  k1_Net1_phos = 1.0;
  k2_Net1_phos = 0.1;
  k3_Net1_dephos = 1.0;
  k4_Net1_dephos = 0.1;
  k1_MEN = 1.0;
  k2_MEN = 0.1;
  k_Bfa1 = 10.0;
  Bfa1_P_threshold = 0.5;
  k1_Bfa1_phos = 1.0;
  k2_Bfa1_phos = 0.1;
  k3_Bfa1_dephos = 1.0;
  k4_Bfa1_dephos = 0.1;
  k_MEN_act = 1.0;
  anaphase_onset = 10.0;
  Separase_0 = 1.0;
  Cdk_threshold = 0.2;
  Cdh1_0 = 1.0;

  // Other declarations:
  var Active_Cdc14, Net1_inhibition, MEN_activity, Bfa1_inhibition, Bfa1_P;
  const cell, k_seq, k_rel, k_deact, k_phos, k_dephos, k_downreg, k_react, k_act, k_inact;
  const k_secdeg, k_deg, k_deg2, k_Net1, Net1_P_threshold, k1_Net1_phos, k2_Net1_phos;
  const k3_Net1_dephos, k4_Net1_dephos, k1_MEN, k2_MEN, k_Bfa1, Bfa1_P_threshold;
  const k1_Bfa1_phos, k2_Bfa1_phos, k3_Bfa1_dephos, k4_Bfa1_dephos, k_MEN_act;
  const anaphase_onset, Separase_0, Cdk_threshold, Cdh1_0;

  // Unit definitions:
  unit substance = 1e-9 mole;
  unit time_unit = 60 second;

  // Display Names:
  cell is "Yeast Cell";
  Cdc14 is "Cdc14 Phosphatase";
  Net1 is "Net1/Cfi1";
  PP2A_Cdc55 is "PP2A-Cdc55 Phosphatase";
  Separase is "Separase/Esp1";
  Securin is "Securin/Pds1";
  Cdk_Clb2 is "Cdk1-Clb2 Kinase";
  Cdc20 is "Cdc20";
  Cdh1 is "Cdh1";
  Polo is "Polo/Cdc5 Kinase";
  Bfa1 is "Bfa1";
  Bub2 is "Bub2";
  MEN is "Mitotic Exit Network";
  Sic1 is "Sic1";
  Active_Cdc14 is "Active Cdc14";
  Net1_P is "Phosphorylated Net1";

end
