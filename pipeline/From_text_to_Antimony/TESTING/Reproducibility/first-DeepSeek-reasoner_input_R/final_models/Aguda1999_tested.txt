  compartment nucleus;

  Rminus3p: iCdc25Ps216_14_3_3 -> iCdc25Ps216 + Fourteen_3_3; nucleus*km3p*iCdc25Ps216_14_3_3;  # v-3'
  R6: -> Fourteen_3_3; nucleus*k6*p53;  # v6
  R13: -> Fourteen_3_3; nucleus*k13;  # v13
  R12: Fourteen_3_3 -> ; nucleus*k12*Fourteen_3_3;  # v12
  // Import and export
  Rin: -> iCdc25; nucleus*kin;  # vin
  Rex: iCdc25Ps216_14_3_3 -> ; nucleus*kex*iCdc25Ps216_14_3_3;  # vex

  // Wee1 subsystem
  R16: -> Wee1; nucleus*k16;  # v16
  R8: Wee1 -> Wee1P; nucleus*(k8*MPF*Wee1 + k8p*Wee1);  # v8
  Rminus8: Wee1P -> Wee1; nucleus*km8*Wee1P;  # v-8
  R17: Wee1P -> ; nucleus*k17*Wee1P;  # v17
