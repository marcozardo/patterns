model *Admon2017_Tumor_Immune_Drug()

  //// Compartments and Species:
  compartment tumor;
  species TI in tumor, TM in tumor, I in tumor, u in tumor;

  //// Reactions:
  R1: TM -> 2 TI; tumor * a4 * TM;
  R2: TI + I -> ; tumor * c1 * TI * I;
  R3: TI -> ; tumor * d2 * TI;
  R4: TI -> TM; tumor * a1 * TI;
  R5: TM -> ; tumor * d3 * TM;
  R6: TM + I -> ; tumor * c3 * TM * I;
  R7: TM -> ; tumor * k1 * (1 - exp(-k2 * u)) * TM;
  R8: -> I; tumor * k;
  R9: -> I; tumor * rho * I * (TI + TM)^n / (alpha + (TI + TM)^n);
  R10: I -> ; tumor * d1 * I;
  R11: I -> ; tumor * k3 * (1 - exp(-k4 * u)) * I;
  R12: u -> ; tumor * gamma_ * u;

  //// Species initializations:
  TI = 1.3;
  TM = 1.2;
  I = 0.9;
  u = 0.05;

  //// Compartment initializations:
  tumor = 1;

  //// Variable initializations:
  a1 = 0.98;
  a4 = 0.8;
  d1 = 0.29;
  d2 = 0.11;
  d3 = 0.4;
  c1 = 0.9;
  c2 = 0.9;
  c3 = 0.085;
  c4 = 0.085;
  k = 0.029;
  k1 = 0.47;
  k2 = 0.57;
  k3 = 0.49;
  k4 = 0.061;
  rho = 0.2;
  alpha = 0.1;
  n = 3;
  gamma_ = 0.85;

  //// Other declarations:
  const tumor, a1, a4, d1, d2, d3, c1, c2, c3, c4, k, k1, k2, k3, k4, rho, alpha, n, gamma_;

  //// Unit definitions:
  // The model uses non-dimensionalized parameters; units are arbitrary.

  //// Display Names:
  tumor is "Tumor microenvironment";
  TI is "Tumor cells in interphase";
  TM is "Tumor cells in mitosis";
  I is "Immune response (CTL)";
  u is "Drug concentration";
  R1 is "Mitosis to interphase (cell division)";
  R2 is "Immune killing of interphase tumor cells";
  R3 is "Natural death of interphase tumor cells";
  R4 is "Interphase to mitosis transition";
  R5 is "Natural death of mitosis tumor cells";
  R6 is "Immune killing of mitosis tumor cells";
  R7 is "Drug killing of mitosis tumor cells";
  R8 is "Constant source of immune cells";
  R9 is "Stimulation of immune cells by tumor";
  R10 is "Natural death of immune cells";
  R11 is "Drug effect on immune cells";
  R12 is "Drug decay";

end
