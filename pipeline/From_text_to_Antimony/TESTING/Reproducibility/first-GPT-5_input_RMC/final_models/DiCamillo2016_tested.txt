  R_IRS1_pY: IRS1 => IRS1_pY; cell*k_IRS1Y*IR_p*IRS1;
  R_IRS1_pY_dephos: IRS1_pY => IRS1; cell*k_dIRS1Y*IRS1_pY;

  R_IRS1_pS_dephos: IRS1_pS => IRS1; cell*k_dIRS1S*IRS1_pS;

  R_IRS1_PI3K_bind: IRS1_pY + PI3K -> IRS1_PI3K; cell*(k8*IRS1_pY*PI3K - k_8*IRS1_PI3K);

  # PI3K-dependent lipid conversion PI(4,5)P2 -> PI(3,4,5)P3 and reverse lipid phosphatases
  R_PIP3_synthesis: PIP2 => PIP3; cell*k9a*IRS1_PI3K*PIP2;
  R_PIP3_degradation: PIP3 => PIP2; cell*k9s*PIP3;

  # PDK1/mTORC2 mediated Akt phosphorylation (Thr and Ser) and formation of doubly phosphorylated Akt
  R_Akt_pT: Akt => Akt_pT; cell*k_Akt_T*PIP3*PDK1*Akt;
  R_Akt_pS: Akt => Akt_pS; cell*k_Akt_S*mTORC2*Akt;
  R_Akt_pp_from_pT: Akt_pT => Akt_pp; cell*k_Akt_S2*mTORC2*Akt_pT;
  R_Akt_pp_from_pS: Akt_pS => Akt_pp; cell*k_Akt_T2*PIP3*PDK1*Akt_pS;

  # AMPK-dependent TSC1/2 Ser phosphorylation (activation) and Akt-dependent Tyr phosphorylation (inhibition)
  R_TSC_Ser_phos: TSC1_2 => TSC1_2_pS; cell*k_TSCSer*AMPK_pT172*TSC1_2;
  R_TSC_Tyr_phos: TSC1_2 => TSC1_2_pY; cell*k_TSCY*(Akt_pT + Akt_pp)*TSC1_2;
  R_TSC_Ser_dephos_by_Akt: TSC1_2_pS => TSC1_2; cell*k_TSCSer_dephos*(Akt_pT + Akt_pp)*TSC1_2_pS;

  # mTORC1 activation by amino acids and inhibition by active TSC1/2; deactivation
  R_mTORC1_activation: mTORC1 => mTORC1_p; cell*k_mTORC1_act*AminoAcids*(1/(1 + K_TSC*TSC1_2_pS))*mTORC1;
  R_mTORC1_deactivation: mTORC1_p => mTORC1; cell*k_mTORC1_deact*mTORC1_p;

  # p70S6K activation by mTORC1_p and deactivation
  R_p70S6K_activation: p70S6K => p70S6K_pT389; cell*k_p70_act*mTORC1_p*p70S6K;
  R_p70S6K_deactivation: p70S6K_pT389 => p70S6K; cell*k_p70_deact*p70S6K_pT389;

  # GLUT4 translocation to membrane activated by Akt phosphorylation (Thr/Ser/both) and internalization
  R_GLUT4_translocation: GLUT4_cyt => GLUT4_memb; cell*k_GLUT4_trans*(Akt_pT + Akt_pS + Akt_pp)*GLUT4_cyt;
  R_GLUT4_internalization: GLUT4_memb -> GLUT4_cyt; cell*k_GLUT4_int*GLUT4_memb;

  # RAS-MAPK pathway: adaptor binding, Ras activation, Raf/MEK/ERK cascade, and ERK->GRB2/SOS feedback
  R_GS_bind: IRS1_pY + GRB2_SOS -> IRS1_GS_complex; cell*(k_gs_on*IRS1_pY*GRB2_SOS - k_gs_off*IRS1_GS_complex);
  R_GS_feedback_dissoc: IRS1_GS_complex => IRS1_pY + GRB2_SOS; cell*(kfb_ERK*ERK_pp + k_gs_off2)*IRS1_GS_complex;

  R_Ras_activation: RasGDP => RasGTP; cell*k_ras_act*IRS1_GS_complex*RasGDP;
  R_Ras_deactivation: RasGTP => RasGDP; cell*k_ras_gap*RasGTP;

  R_Raf_activation: Raf => Raf_p; cell*k_raf_act*RasGTP*Raf;
  R_Raf_inactivation: Raf_p => Raf; cell*k_raf_inact*Raf_p;

  R_MEK_activation: MEK => MEK_pp; cell*k_mek_act*Raf_p*MEK;
  R_MEK_deactivation: MEK_pp => MEK; cell*k_mek_deact*MEK_pp;

  R_ERK_activation: ERK => ERK_pp; cell*k_erk_act*MEK_pp*ERK;
  R_ERK_deactivation: ERK_pp => ERK; cell*k_erk_deact*ERK_pp;
