  species C in BMU, B in BMU, T in BMU, J in BMU;

  //// Reactions:
  R_C_prod:  => C; BMU*(alpha_C*(1 + h_CT*T)*C^g_CC*B^g_CB);
  R_C_decay: C => ; BMU*(beta_C*C);
  R_B_prod:  => B; BMU*(alpha_B*(1 - h_BT*T)*C^g_BC*B^g_BB);
  R_B_decay: B => ; BMU*(beta_B*B);
  R_T_prod:  => T; BMU*(alpha_T*C^g_TC*T^g_TT);
  R_T_decay: T => ; BMU*(beta_T*T);
  R_fusion:  C + T => J; BMU*(alpha_J*C*T);        # osteoclast?myeloma fusion
  R_J_decay: J => ; BMU*(beta_J*J);                # joint-cell loss (apoptosis + non-productive loss)
  R_J_diss:  J => T; BMU*(kappa*J);                # joint-cell dissociation yielding a myeloma cell

  C = 2;        # osteoclasts per BMU (MGUS-scale; ~2)
