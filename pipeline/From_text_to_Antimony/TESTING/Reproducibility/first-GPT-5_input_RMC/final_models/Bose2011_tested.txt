model *Bose2011_TumorImmune_NoiseMeanField()

  //// Compartments and Species:
  compartment system;
  species $x in system, $y in system, $z in system;

  //// Assignment Rules:
  gamma_x := 1 + R*(1 - Dx*tau) + 0.5*Dx*Dz*tau;
  beta_zx := Dx*(1 + R*tau);
  alpha_xz := R*(1 - Dx*tau) + Dx*Dz*tau;
  mu_eff := mu - delta_mu_noise;

  //// Rate Rules:
  x' = gamma_x*x - x^2 - x*y + beta_zx*z;
  y' = y*z - (rho - Dy)*y;
  z' = alpha_xz*x - mu_eff*z;

  //// Reactions:
  # none (dynamics defined by rate rules)

  //// Events:
  # none

  //// Species initializations:
  x = 1e-6;
  y = 1e-3;
  z = 0;

  //// Compartment initializations:
  system = 1;

  //// Variable initializations:
  R = 1.0;
  tau = 2.0;
  Dx = 2.1;
  Dy = 0.01;
  Dz = 1.2;
  rho = 0.06;
  mu = 20;
  delta_mu_noise = 0;

  //// Other declarations:
  var gamma_x, beta_zx, alpha_xz, mu_eff;
  const system, R, tau, Dx, Dy, Dz, rho, mu, delta_mu_noise;

  # omitted (model is in dimensionless variables per the paper's scaling)

  system is "Well-mixed compartment (dimensionless volume)";
