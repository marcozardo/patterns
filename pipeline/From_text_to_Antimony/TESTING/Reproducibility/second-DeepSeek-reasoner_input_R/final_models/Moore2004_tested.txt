model *Moore2004_CML_Tcell()

compartment blood;
species T_n in blood, T_e in blood, C in blood;

//// Assignment Rules:

//// Rate Rules:

T_e' = alpha_n*k_n*T_n*(C/(C + theta)) + alpha_e*T_e*(C/(C + theta)) - d_e*T_e - gamma_e*C*T_e;
C' = r_c*C*log(C_max/C) - d_c*C - gamma_c*C*T_e;

//// Reactions:

//// Events:

theta = 100;

alpha_e = 0.2;
C_max = 300000;
r_c = 0.03;
gamma_e = 0.005;
gamma_c = 0.005;

//// Other declarations:
var T_n, T_e, C;
const blood, s_n, d_n, d_e, d_c, k_n, theta, alpha_n, alpha_e, C_max, r_c, gamma_e, gamma_c;

unit concentration = 1e-6 liter^-1;

gamma_c is "Rate constant for CML loss due to effector T cell encounters";
