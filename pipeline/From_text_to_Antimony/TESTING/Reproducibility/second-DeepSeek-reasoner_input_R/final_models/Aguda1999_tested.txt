model *Aguda1999_G2_DNA_damage_checkpoint()

  //// Compartments and Species:
  compartment cell;
  species $Chk1 in cell;  // boundary species defined by assignment rule
  species Chk1P in cell, Rad3 in cell, p53 in cell, preMPF in cell, MPF in cell, p21 in cell, p21_MPF in cell;
  species iCdc25 in cell, iCdc25Ps216 in cell, iCdc25Ps216_14_3_3 in cell, aCdc25 in cell, aCdc25Ps216 in cell;
  species Wee1 in cell, Wee1P in cell;
  species X14_3_3 in cell;  // 14-3-3 protein

  //// Assignment Rules:
  Chk1 := Chk1_total - Chk1P;

  //// Rate Rules:
  // none ? all dynamics are captured by reactions

  //// Reactions:
  R1: Chk1 -> Chk1P; k1*Chk1*Rad3;
  Rminus1: Chk1P -> Chk1; km1*Chk1P;

  R2: aCdc25 -> aCdc25Ps216; (k2*Chk1P + kctak2)*aCdc25;
  Rminus2: aCdc25Ps216 -> aCdc25; km2*aCdc25Ps216;

  R2prime: iCdc25 -> iCdc25Ps216; (k2prime*Chk1P + kctak2prime)*iCdc25;

  R3prime: iCdc25Ps216 + X14_3_3 -> iCdc25Ps216_14_3_3; k3prime*iCdc25Ps216*X14_3_3;
  Rminus3prime: iCdc25Ps216_14_3_3 -> iCdc25Ps216 + X14_3_3; km3prime*iCdc25Ps216_14_3_3;

  R4:  -> Rad3; k4;
  R5: Rad3 -> ; k5*Rad3;

  R6:  -> X14_3_3; k6*p53;
  R7: iCdc25 -> aCdc25; (k7*MPF + kPlk1)*iCdc25;
  Rminus7: aCdc25 -> iCdc25; km7*aCdc25;

  R8: Wee1 -> Wee1P; (k8*MPF + k8prime)*Wee1;
  Rminus8: Wee1P -> Wee1; km8*Wee1P;

  R9: preMPF -> MPF; (k9*(aCdc25 + aCdc25Ps216) + k9prime)*preMPF;
  Rminus9: MPF -> preMPF; km9*MPF*Wee1;

  R10:  -> p53; k10;
  R11: p53 -> ; k11*p53;

  R12: X14_3_3 -> ; k12*X14_3_3;
  R13:  -> X14_3_3; k13;

  R14:  -> preMPF; k14/(1 + k14prime*p53);
  R15: MPF -> ; k15*MPF*MPF;

  R16:  -> Wee1; k16;
  R17: Wee1P -> ; k17*Wee1P;

  R18: iCdc25Ps216 -> aCdc25Ps216; (k18*MPF + kPlk1prime)*iCdc25Ps216;
  Rminus18: aCdc25Ps216 -> iCdc25Ps216; km18*aCdc25Ps216;

  R20:  -> p21; k20*p53;
  R21:  -> p21; k21;
  R22: p21 -> ; k22*p21;

  R23: MPF + p21 -> p21_MPF; k23*MPF*p21;
  Rminus23: p21_MPF -> MPF + p21; km23*p21_MPF;

  Export: iCdc25Ps216_14_3_3 -> ; kex*iCdc25Ps216_14_3_3;
  Import:  -> iCdc25; kin;

  //// Events:
  // none

  //// Species initializations:
  Chk1P = 1e-3;
  Rad3 = 0;
  p53 = 0;
  preMPF = 1e-3;
  MPF = 1e-4;
  p21 = 0;
  p21_MPF = 0;
  iCdc25 = 1e-3;
  iCdc25Ps216 = 2e-3;
  iCdc25Ps216_14_3_3 = 0.03;
  aCdc25 = 1e-3;
  aCdc25Ps216 = 0;
  Wee1 = 1e-2;
  Wee1P = 0;
  X14_3_3 = 2;

  //// Compartment initializations:
  cell = 1;

  //// Variable initializations:
  // Rate constants (dimensionless)
  k1 = 1.0;
  km1 = 0.1;
  k2 = 0.1;
  kctak2 = 0;
  km2 = 0.1;
  k2prime = 0;        // set to zero as per Fig.8 legend
  kctak2prime = 0;
  k3prime = 0.1;
  km3prime = 0.1;
  k4 = 0;             // no DNA damage in default simulation
  k5 = 1.0;
  k6 = 0.1;
  k7 = 1.0;
  kPlk1 = 0;
  km7 = 1.0;
  k8 = 1.0;
  k8prime = 0.01;
  km8 = 1.0;
  k9 = 1.0;
  k9prime = 0.01;
  km9 = 1.0;
  k10 = 0;            // no DNA damage in default simulation
  k11 = 1.0;
  k12 = 0.1;

  k14 = 0.1;

  k15 = 0.1;
  k16 = 1.0;
  k17 = 1.0;
  k18 = 1.0;

  km18 = 1.0;
  k20 = 1.0;
  k21 = 1.0;
  k22 = 0.1;
  k23 = 1.0;
  km23 = 1.0;
  kex = 1.0;
  kin = 1e-3;
  Chk1_total = 1.0;

  //// Other declarations:
  const k1, km1, k2, kctak2, km2, k2prime, kctak2prime, k3prime, km3prime, k4, k5, k6, k7, kPlk1, km7, k8, k8prime, km8, k9, k9prime, km9, k10, k11, k12, k13, k14, k14prime, k15, k16, k17, k18, kPlk1prime, km18, k20, k21, k22, k23, km23, kex, kin, Chk1_total;
