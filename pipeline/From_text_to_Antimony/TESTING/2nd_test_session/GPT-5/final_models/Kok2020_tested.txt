model *Kok2020_IFNa_Sensitization()

  // Compartments and Species:
  compartment extracellular, membrane, cytoplasm, nucleus;
  species IFN in extracellular;

  species STAT1c in cytoplasm, STAT2c in cytoplasm, IRF9c in cytoplasm;
  species pSTAT1dimc in cytoplasm, pSTAT1pSTAT2c in cytoplasm, ISGF3c in cytoplasm;
  species STAT1n in nucleus, STAT2n in nucleus, IRF9n in nucleus;
  species pSTAT1dimn in nucleus, pSTAT1pSTAT2n in nucleus, ISGF3n in nucleus;

  // Assignment Rules:

  OccISREbs := kISRE*ISGF3n;                                              # ISRE occupancy by ISGF3
  OccGAS_ISRE := kGASISRE*(pSTAT1pSTAT2n + ISGF3n);                       # Combined GAS+ISRE occupancy by heterodimer and ISGF3
  Tx_SOCS1_eff := k_tx_SOCS1*OccGAS_ISRE/(1 + k_irf2*IRF2n);              # IRF2-mediated repression on SOCS1 transcription

  // Reactions:
  # Ligand-receptor interaction and receptor activation/inactivation

  R_aRec_deg: aRecIFN -> ; (k_deg_aRec + k_soc1*SOCS1 + k_soc1_usp18*SOCS1*USP18)*aRecIFN;

  # Receptor-driven phosphorylation/dimerization and ISGF3 assembly (approximated single-step formations)
  R_pSTAT1dim_c: 2 STAT1c -> pSTAT1dimc; k_p1*aRecIFN*STAT1c^2;
  R_pSTAT1_2_c: STAT1c + STAT2c -> pSTAT1pSTAT2c; k_p12*aRecIFN*STAT1c*STAT2c;
  R_ISGF3_c: IRF9c + pSTAT1pSTAT2c -> ISGF3c; k_isgf3*pSTAT1pSTAT2c*IRF9c - k_dis_isgf3*ISGF3c;

  # Nuclear-cytoplasmic shuttling of complexes
  R_imp_p11: pSTAT1dimc -> pSTAT1dimn; k_imp_p11*pSTAT1dimc;
  R_exp_p11: pSTAT1dimn -> pSTAT1dimc; k_exp_p11*pSTAT1dimn;
  R_imp_p12: pSTAT1pSTAT2c -> pSTAT1pSTAT2n; k_imp_p12*pSTAT1pSTAT2c;
  R_exp_p12: pSTAT1pSTAT2n -> pSTAT1pSTAT2c; k_exp_p12*pSTAT1pSTAT2n;
  R_imp_ISGF3: ISGF3c -> ISGF3n; k_imp_isgf3*ISGF3c;
  R_exp_ISGF3: ISGF3n -> ISGF3c; k_exp_isgf3*ISGF3n;

  # Dissociation of nuclear complexes to free components
  R_diss_p11_n: pSTAT1dimn -> 2 STAT1n; k_diss_p11n*pSTAT1dimn;
  R_diss_p12_n: pSTAT1pSTAT2n -> STAT1n + STAT2n; k_diss_p12n*pSTAT1pSTAT2n;
  R_diss_ISGF3_n: ISGF3n -> STAT1n + STAT2n + IRF9n; k_diss_isgf3n*ISGF3n;

  # Shuttling of free STAT1/STAT2/IRF9
  R_imp_STAT1: STAT1c -> STAT1n; k_imp_s1*STAT1c;
  R_exp_STAT1: STAT1n -> STAT1c; k_exp_s1*STAT1n;
  R_imp_STAT2: STAT2c -> STAT2n; k_imp_s2*STAT2c;
  R_exp_STAT2: STAT2n -> STAT2c; k_exp_s2*STAT2n;
  R_imp_IRF9: IRF9c -> IRF9n; k_imp_irf9*IRF9c;
  R_exp_IRF9: IRF9n -> IRF9c; k_exp_irf9*IRF9n;

  # Transcription (driven by promoter occupancies)
  TX_IRF9:  => IRF9mRNA; k_tx_IRF9*OccGAS_ISRE;
  TX_STAT1:  => STAT1mRNA; k_tx_STAT1*OccGAS_ISRE;
  TX_STAT2:  => STAT2mRNA; k_tx_STAT2*OccGAS_ISRE;
  TX_USP18:  => USP18mRNA; k_tx_USP18*OccGAS_ISRE;
  TX_SOCS1:  => SOCS1mRNA; Tx_SOCS1_eff;
  TX_SOCS3:  => SOCS3mRNA; k_tx_SOCS3*OccGASbs;
  TX_IRF2:   => IRF2mRNA; k_tx_IRF2*OccGAS_ISRE;

  # mRNA degradation
  DEG_IRF9m: IRF9mRNA => ; k_deg_m_IRF9*IRF9mRNA;
  DEG_STAT1m: STAT1mRNA => ; k_deg_m_STAT1*STAT1mRNA;
  DEG_STAT2m: STAT2mRNA => ; k_deg_m_STAT2*STAT2mRNA;
  DEG_USP18m: USP18mRNA => ; k_deg_m_USP18*USP18mRNA;
  DEG_SOCS1m: SOCS1mRNA => ; k_deg_m_SOCS1*SOCS1mRNA;
  DEG_SOCS3m: SOCS3mRNA => ; k_deg_m_SOCS3*SOCS3mRNA;
  DEG_IRF2m: IRF2mRNA => ; k_deg_m_IRF2*IRF2mRNA;

  # Translation
  TL_IRF9: IRF9mRNA => IRF9c; k_tl_IRF9*IRF9mRNA;
  TL_STAT1: STAT1mRNA => STAT1c; k_tl_STAT1*STAT1mRNA;
  TL_STAT2: STAT2mRNA => STAT2c; k_tl_STAT2*STAT2mRNA;
  TL_USP18: USP18mRNA => USP18; k_tl_USP18*USP18mRNA;

  TL_IRF2: IRF2mRNA => IRF2n; k_tl_IRF2*IRF2mRNA;

  DEG_STAT2c: STAT2c => ; k_deg_STAT2*STAT2c;
  DEG_IRF9c: IRF9c => ; k_deg_IRF9*IRF9c;
  DEG_USP18: USP18 => ; k_deg_USP18*USP18;
  DEG_SOCS1: SOCS1 => ; k_deg_SOCS1*SOCS1;
  DEG_SOCS3: SOCS3 => ; k_deg_SOCS3*SOCS3;
  DEG_IRF2n: IRF2n => ; k_deg_IRF2*IRF2n;

  E1_stim:    at time >= t_stim: IFN = IFN_stim;
