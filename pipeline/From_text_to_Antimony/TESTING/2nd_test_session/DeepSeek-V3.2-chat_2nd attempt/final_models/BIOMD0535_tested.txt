  STAT3_phosphorylation_GI_reaction: STAT3_GI -> pSTAT3_GI; kRAct * IL6_GI * STAT3_GI;
  pSTAT3_dephosphorylation_GI_reaction: pSTAT3_GI -> STAT3_GI; VmnDephos * pSTAT3_GI / (KrmDephos + pSTAT3_GI);
  IL6_Liver_synthesis: $EmptySet_Liver => IL6_Liver; ksynfIL6Liver;
  IL6_Liver_degradation: IL6_Liver => $EmptySet_Liver; kdegIL6 * IL6_Liver;
  STAT3_phosphorylation_Liver_reaction: STAT3_Liver -> pSTAT3_Liver; kRAct * IL6_Liver * STAT3_Liver;
  pSTAT3_dephosphorylation_Liver_reaction: pSTAT3_Liver -> STAT3_Liver; VmnDephos * pSTAT3_Liver / (KrmDephos + pSTAT3_Liver);
  Anti_IL6_binding: IL6 + Anti_IL6 -> IL6_Anti_IL6_complex; kon_Anti_IL6 * IL6 * Anti_IL6 - koff_Anti_IL6 * IL6_Anti_IL6_complex;
  Anti_IL6R_binding: sIL6R + Anti_IL6R -> sIL6R_Anti_IL6R_complex; kon_Anti_IL6R * sIL6R * Anti_IL6R - koff_Anti_IL6R * sIL6R_Anti_IL6R_complex;
  Anti_IL6_complex_binding: IL6_sIL6R_complex + Anti_IL6 -> IL6_sIL6R_Anti_IL6_complex; kon_Anti_IL6_complex * IL6_sIL6R_complex * Anti_IL6 - koff_Anti_IL6_complex * IL6_sIL6R_Anti_IL6_complex;
  sgp130Fc_binding: IL6_sIL6R_complex + sgp130Fc -> IL6_sIL6R_sgp130Fc_complex; kon_sgp130Fc * IL6_sIL6R_complex * sgp130Fc - koff_sgp130Fc * IL6_sIL6R_sgp130Fc_complex;

  // Events:

  Anti_IL6R_dosing: at time >= 0: Anti_IL6R = Anti_IL6R + dose_Anti_IL6R;
