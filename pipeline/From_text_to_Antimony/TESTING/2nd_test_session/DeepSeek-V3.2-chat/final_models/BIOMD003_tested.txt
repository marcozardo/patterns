model *Goldbeter1991_Mitotic_Oscillator()
  // Compartments and Species:
  compartment cell;
  species C in cell, M in cell, X in cell;
  species M_inactive in cell, X_inactive in cell;

  // Assignment Rules:
  V1 := (C/(Kc + C))*VM1;
  V3 := M*VM3;

  // Reactions:
  Cyclin_synthesis: => C; cell*vi;
  Cyclin_degradation: C => ; cell*(vd*X*C/(Kd + C));
  Cyclin_nonspecific_degradation: C => ; cell*kd*C;
  cdc2_activation: M_inactive => M; cell*(V1*(1-M)/(K1 + (1-M)));
  cdc2_inactivation: M => M_inactive; cell*(V2*M/(K2 + M));
  Protease_activation: X_inactive => X; cell*(V3*(1-X)/(K3 + (1-X)));
  Protease_inactivation: X => X_inactive; cell*(V4*X/(K4 + X));

  // Species initializations:
  C = 0;
  M = 0;
  X = 0;
  M_inactive = 1;
  X_inactive = 1;

  // Compartment initializations:
  cell = 1;

  // Variable initializations:
  vi = 0.023;
  vd = 0.1;
  kd = 3.33e-3;
  Kd = 0.02;
  Kc = 0.3;
  VM1 = 0.5;
  V2 = 0.167;
  VM3 = 0.2;
  V4 = 0.1;
  K1 = 0.1;
  K2 = 0.1;
  K3 = 0.1;
  K4 = 0.1;

  // Other declarations:
  var V1, V3;
  const cell, vi, vd, kd, Kd, Kc, VM1, V2, VM3, V4, K1, K2, K3, K4;

  // Unit definitions:
  unit volume = 1e-3 litre;
  unit time_unit = 60 second;
  unit substance = 1e-6 mole;

  // Display Names:
  cell is "Cell";
  C is "Cyclin";
  M is "Active cdc2 kinase";
  X is "Active cyclin protease";
  M_inactive is "Inactive cdc2 kinase";
  X_inactive is "Inactive cyclin protease";
  vi is "Cyclin synthesis rate";
  vd is "Maximum cyclin degradation rate";
  kd is "Non-specific degradation rate";
  Kd is "Michaelis constant for cyclin degradation";
  Kc is "Michaelis constant for cyclin activation";
  VM1 is "Maximum phosphatase rate";
  V2 is "cdc2 kinase inactivation rate";
  VM3 is "Maximum protease activation rate";
  V4 is "Protease inactivation rate";
  Cyclin_synthesis is "Cyclin synthesis";
  Cyclin_degradation is "Cyclin degradation";
  Cyclin_nonspecific_degradation is "Non-specific cyclin degradation";
  cdc2_activation is "cdc2 kinase activation";
  cdc2_inactivation is "cdc2 kinase inactivation";
  Protease_activation is "Protease activation";
  Protease_inactivation is "Protease inactivation";
end
