model *Dwivedi2014_Multiscale_IL6_Crohns_Model()

  // Compartments and Species:
  compartment serum, liver, GI_tract;
  species IL6 in serum, sIL6R in serum, CRP in serum, sgp130 in serum;
  species IL6_sIL6R_complex in serum, IL6_sIL6R_sgp130_complex in serum;
  species IL6R_membrane in liver, IL6R_membrane in GI_tract;
  species gp130 in liver, gp130 in GI_tract;
  species STAT3 in liver, STAT3 in GI_tract;
  species pSTAT3 in liver, pSTAT3 in GI_tract;
  species pSTAT3_dimer in liver, pSTAT3_dimer in GI_tract;
  species SOCS in liver, SOCS in GI_tract;
  species Gene_products in liver, Gene_products in GI_tract;
  species Anti_IL6 in serum, Anti_IL6R in serum, sgp130Fc in serum;

  // Assignment Rules:
  IL6_total := IL6 + IL6_sIL6R_complex + IL6_sIL6R_sgp130_complex;
  sIL6R_total := sIL6R + IL6_sIL6R_complex + IL6_sIL6R_sgp130_complex;
  sgp130_total := sgp130 + IL6_sIL6R_sgp130_complex;

  // Reactions:
  // IL-6 Signaling Reactions
  IL6_synthesis_GI: => IL6; k_synth_IL6_GI*GI_tract;
  IL6_synthesis_liver: => IL6; k_synth_IL6_liver*liver;
  IL6_degradation: IL6 => ; k_deg_IL6*IL6;
  
  sIL6R_shedding: IL6R_membrane => sIL6R; k_shed_sIL6R*IL6R_membrane*CRP;
  sIL6R_degradation: sIL6R => ; k_deg_sIL6R*sIL6R;
  
  Complex_formation: IL6 + sIL6R -> IL6_sIL6R_complex; kf_complex*IL6*sIL6R - kr_complex*IL6_sIL6R_complex;
  Trans_signaling: IL6_sIL6R_complex + gp130 -> IL6_sIL6R_sgp130_complex; kf_trans*IL6_sIL6R_complex*gp130;
  Classical_signaling: IL6 + IL6R_membrane -> IL6_sIL6R_complex; kf_classical*IL6*IL6R_membrane;
  
  sgp130_inhibition: IL6_sIL6R_complex + sgp130 -> IL6_sIL6R_sgp130_complex; kf_inhibit*IL6_sIL6R_complex*sgp130;
  
  STAT3_phosphorylation: STAT3 -> pSTAT3; k_phosphorylation*STAT3*(IL6_sIL6R_complex + IL6_sIL6R_sgp130_complex);
  STAT3_dephosphorylation: pSTAT3 -> STAT3; k_dephosphorylation*pSTAT3;
  Dimer_formation: pSTAT3 + pSTAT3 -> pSTAT3_dimer; kf_dimer*pSTAT3*pSTAT3;
  
  CRP_synthesis: => CRP; k_CRP_synthesis*pSTAT3_dimer;
  CRP_degradation: CRP => ; k_deg_CRP*CRP;

  // Therapeutic Antibody Reactions
  Anti_IL6_binding: IL6 + Anti_IL6 -> IL6_Anti_IL6; kf_anti_IL6*IL6*Anti_IL6 - kr_anti_IL6*IL6_Anti_IL6;
  Anti_IL6R_binding: sIL6R + Anti_IL6R -> sIL6R_Anti_IL6R; kf_anti_IL6R*sIL6R*Anti_IL6R - kr_anti_IL6R*sIL6R_Anti_IL6R;
  Anti_IL6_complex_binding: IL6_sIL6R_complex + Anti_IL6 -> IL6_sIL6R_Anti_IL6; kf_anti_complex*IL6_sIL6R_complex*Anti_IL6;
  
  sgp130Fc_binding: IL6_sIL6R_complex + sgp130Fc -> IL6_sIL6R_sgp130Fc; kf_sgp130Fc*IL6_sIL6R_complex*sgp130Fc;

  // Species initializations:
  IL6 = 0.042;  // 1 pg/ml converted to pmol/l
  sIL6R = 1.6;  // 80 ng/ml converted to nmol/l
  CRP = 0.04;  // 1 mg/l converted to ?mol/l
  sgp130 = 2.8;  // 280 ng/ml converted to nmol/l
  STAT3 = 1;
  pSTAT3 = 1;
  pSTAT3_dimer = 0;
  SOCS = 0;
  Gene_products = 0;
  Anti_IL6 = 0;
  Anti_IL6R = 0;
  sgp130Fc = 0;
  IL6_sIL6R_complex = 0;
  IL6_sIL6R_sgp130_complex = 0;
  IL6_Anti_IL6 = 0;
  sIL6R_Anti_IL6R = 0;
  IL6_sIL6R_Anti_IL6 = 0;
  IL6_sIL6R_sgp130Fc = 0;

  // Compartment initializations:
  serum = 3.0;
  liver = 1.5;
  GI_tract = 1.0;

  // Variable initializations:
  k_synth_IL6_GI = 0.1;
  k_synth_IL6_liver = 0.01;
  k_deg_IL6 = 0.1;
  k_shed_sIL6R = 0.05;
  k_deg_sIL6R = 0.01;
  kf_complex = 1.0;
  kr_complex = 0.1;
  kf_trans = 0.5;
  kf_classical = 0.5;
  kf_inhibit = 0.8;
  k_phosphorylation = 0.2;
  k_dephosphorylation = 0.1;
  kf_dimer = 0.3;
  k_CRP_synthesis = 0.15;
  k_deg_CRP = 0.05;
  k_SOCS_synthesis = 0.1;
  k_SOCS_degradation = 0.05;
  k_gene_synthesis = 0.08;
  k_gene_degradation = 0.04;
  
  // Antibody parameters
  kf_anti_IL6 = 1.0;
  kr_anti_IL6 = 0.1;
  kf_anti_IL6R = 1.0;
  kr_anti_IL6R = 0.1;
  kf_anti_complex = 1.0;
  kf_sgp130Fc = 0.8;

  // Other declarations:
  var IL6_total, sIL6R_total, sgp130_total;
  const serum, liver, GI_tract;

  // Unit definitions:
  unit volume = 1 litre;
  unit time_unit = 3600 second;
  unit concentration = 1e-9 mole/litre;

  // Display Names:
  serum is "Blood Serum";

  IL6_sIL6R_sgp130_complex is "IL-6/sIL-6R/sgp130 Complex";
  Anti_IL6 is "Anti-IL-6 Antibody";
  Anti_IL6R is "Anti-IL-6R Antibody";
  sgp130Fc is "sgp130-Fc Fusion Protein";
  pSTAT3 is "Phosphorylated STAT3";
  SOCS is "Suppressor of Cytokine Signaling";
end
