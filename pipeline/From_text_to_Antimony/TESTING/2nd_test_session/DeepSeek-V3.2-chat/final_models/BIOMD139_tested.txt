model *Hoffmann2002_NFkB_Signaling()
  // Compartments and Species:
  compartment cytoplasm, nucleus;
  species NFkB in cytoplasm, NFkB_n in nucleus;
  species IkBa in cytoplasm, IkBb in cytoplasm, IkBe in cytoplasm;
  species IKK in cytoplasm;
  species TNF in cytoplasm;
  species IP10_gene in nucleus, RANTES_gene in nucleus;
  species IP10_mRNA in cytoplasm, RANTES_mRNA in cytoplasm;

  // Assignment Rules:
  IkBa_synthesis := k_syn_ba * NFkB_n;
  IkBb_synthesis := k_syn_bb * NFkB_n;
  IkBe_synthesis := k_syn_be * NFkB_n;

  // Rate Rules:
  IKK' = k_IKK_act * TNF_stimulus - k_IKK_deact * IKK;
  NFkB_n' = k_transport * NFkB - k_export * NFkB_n - k_binding * NFkB_n * (IkBa + IkBb + IkBe);

  // Reactions:
  IkBa_degradation: IkBa => ; k_deg_ba * IKK * IkBa;
  IkBb_degradation: IkBb => ; k_deg_bb * IKK * IkBb;
  IkBe_degradation: IkBe => ; k_deg_be * IKK * IkBe;
  NFkB_complex_formation: NFkB + IkBa => NFkB_IkBa; k_assoc_ba * NFkB * IkBa;
  NFkB_release: NFkB_IkBa => NFkB + IkBa; k_dissoc_ba * NFkB_IkBa;
  IP10_transcription:  => IP10_mRNA; k_trans_IP10 * NFkB_n * IP10_gene;
  RANTES_transcription:  => RANTES_mRNA; k_trans_RANTES * NFkB_n * RANTES_gene;
  mRNA_degradation: IP10_mRNA => ; k_deg_mRNA * IP10_mRNA;
  mRNA_degradation2: RANTES_mRNA => ; k_deg_mRNA * RANTES_mRNA;

  // Events:
  at (time > stimulus_duration): TNF_stimulus = 0;

  // Species initializations:
  NFkB = 100;
  NFkB_n = 0;
  IkBa = 50;
  IkBb = 30;
  IkBe = 20;
  IKK = 0;
  TNF = 10;
  IP10_gene = 1;
  RANTES_gene = 1;
  IP10_mRNA = 0;
  RANTES_mRNA = 0;

  // Compartment initializations:
  cytoplasm = 1;
  nucleus = 1;

  // Variable initializations:

  k_IKK_deact = 0.05;
  k_transport = 0.1;
  k_export = 0.05;
  k_binding = 0.01;
  k_deg_ba = 0.1;
  k_deg_bb = 0.08;
  k_deg_be = 0.06;
  k_assoc_ba = 0.01;
  k_dissoc_ba = 0.005;
  k_syn_ba = 0.02;
  k_syn_bb = 0.015;
  k_syn_be = 0.01;
  k_trans_IP10 = 0.1;
  k_trans_RANTES = 0.05;
  k_deg_mRNA = 0.1;
  TNF_conc = 10;
  stimulus_duration = 60;
  TNF_stimulus = piecewise(TNF_conc, time < stimulus_duration, 0);

  // Other declarations:
  var IkBa_synthesis, IkBb_synthesis, IkBe_synthesis;
  const cytoplasm, nucleus, k_IKK_act, k_IKK_deact, k_transport, k_export, k_binding;
  const k_deg_ba, k_deg_bb, k_deg_be, k_assoc_ba, k_dissoc_ba, k_syn_ba, k_syn_bb;
  const k_syn_be, k_trans_IP10, k_trans_RANTES, k_deg_mRNA, TNF_conc, stimulus_duration, TNF_stimulus;

  // Unit definitions:
  unit volume = 1e-3 litre;
  unit time_unit = 60 second;
  unit substance = 1e-9 mole;

  // Display Names:
  volume is "microliter";
  time_unit is "minute";
  substance is "nanomole";
  cytoplasm is "Cytoplasm";
  nucleus is "Nucleus";
  NFkB is "NF-kB cytoplasmic";
