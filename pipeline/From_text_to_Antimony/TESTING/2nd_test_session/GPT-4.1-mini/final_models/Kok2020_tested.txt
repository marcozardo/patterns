model *Kok2020_IFNa_Signal_Transduction()

  // Compartments and Species:
  compartment cytoplasm, nucleus;
  species IFNa in cytoplasm, Rec in cytoplasm, aRecIFN in cytoplasm;
  species STAT1c in cytoplasm, STAT2c in cytoplasm, IRF9c in cytoplasm;
  species pSTAT1dimc in cytoplasm, pSTAT1pSTAT2c in cytoplasm, ISGF3c in cytoplasm;
  species SOCS3mRNA in cytoplasm, USP18mRNA in cytoplasm, SOCS1mRNA in cytoplasm, IRF2mRNA in cytoplasm;
  species SOCS3 in cytoplasm, USP18 in cytoplasm, SOCS1 in cytoplasm, IRF2 in cytoplasm;
  species STAT1n in nucleus, STAT2n in nucleus, IRF9n in nucleus;
  species pSTAT1dimn in nucleus, pSTAT1pSTAT2n in nucleus, ISGF3n in nucleus;
  species OccGASbs in nucleus, OccISREbs in nucleus, OccGASbs_ISREbs in nucleus;
  species IRF9mRNA in cytoplasm, STAT1mRNA in cytoplasm, STAT2mRNA in cytoplasm;

  // Assignment Rules:
  m_inf_STAT1 := alpha_m_STAT1/(alpha_m_STAT1 + beta_m_STAT1);
  alpha_m_STAT1 := -0.1*(V + 35)/(exp(-0.1*(V + 35)) - 1);
  beta_m_STAT1 := 4*exp(-(V + 60)/18);
  alpha_h_STAT1 := 0.07*exp(-(V + 58)/20);
  beta_h_STAT1 := 1/(exp(-0.1*(V + 28)) + 1);
  m_inf_STAT2 := alpha_m_STAT2/(alpha_m_STAT2 + beta_m_STAT2);
  alpha_m_STAT2 := -0.1*(V + 35)/(exp(-0.1*(V + 35)) - 1);
  beta_m_STAT2 := 4*exp(-(V + 60)/18);
  alpha_h_STAT2 := 0.07*exp(-(V + 58)/20);
  beta_h_STAT2 := 1/(exp(-0.1*(V + 28)) + 1);
  alpha_n_STAT1 := -0.01*(V + 34)/(exp(-0.1*(V + 34)) - 1);
  beta_n_STAT1 := 0.125*exp(-(V + 44)/80);
  F := 1/(1 + exp(-(V_pre - theta_syn)/2));
  tau_0 := Cm/gL;
  I_Na := gNa*m_inf_STAT1^3*h*(V - E_Na);
  I_K := gK*n^4*(V - E_K);
  I_L := gL*(V - E_L);
  I_syn := g_syn*s*(V - E_syn);

  // Rate Rules:
  h' = phi*(alpha_h_STAT1*(1 - h) - beta_h_STAT1*h);
  n' = phi*(alpha_n_STAT1*(1 - n) - beta_n_STAT1*n);
  s' = alpha*F*(1 - s) - beta*s;
  V' = (I_app - (I_Na + I_K + I_L + I_syn))/Cm;

  // Reactions:
  R1: IFNa + Rec -> aRecIFN; k1*IFNa*Rec;
  R2: aRecIFN + STAT1c -> pSTAT1dimc; k2*aRecIFN*STAT1c;
  R3: aRecIFN + STAT2c -> pSTAT1pSTAT2c; k3*aRecIFN*STAT2c;
  R4: pSTAT1pSTAT2c + IRF9c -> ISGF3c; k4*pSTAT1pSTAT2c*IRF9c;
  R5: pSTAT1dimc -> STAT1n; k5*pSTAT1dimc;
  R6: pSTAT1pSTAT2c -> STAT2n; k6*pSTAT1pSTAT2c;
  R7: ISGF3c -> ISGF3n; k7*ISGF3c;
  R8: ISGF3n -> SOCS3mRNA; k8*ISGF3n;
  R9: ISGF3n -> USP18mRNA; k9*ISGF3n;
  R10: ISGF3n -> SOCS1mRNA; k10*ISGF3n;
  R11: ISGF3n -> IRF2mRNA; k11*ISGF3n;
  R12: SOCS3mRNA -> SOCS3; k12*SOCS3mRNA;
  R13: USP18mRNA -> USP18; k13*USP18mRNA;
  R14: SOCS1mRNA -> SOCS1; k14*SOCS1mRNA;
  R15: IRF2mRNA -> IRF2; k15*IRF2mRNA;
  R16: SOCS1 + aRecIFN -> ; k16*SOCS1*aRecIFN;  # SOCS1-mediated degradation
  R17: USP18 + SOCS1 + aRecIFN -> ; k17*USP18*SOCS1*aRecIFN;  # USP18 acts as cofactor for SOCS1
  R18: STAT1n -> STAT1c; k18*STAT1n;
  R19: STAT2n -> STAT2c; k19*STAT2n;
  R20: IRF9n -> IRF9c; k20*IRF9n;
