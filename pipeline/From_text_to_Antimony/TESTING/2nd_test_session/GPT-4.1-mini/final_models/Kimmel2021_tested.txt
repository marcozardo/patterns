model *Kimmel2021_CAR_T_cell_therapy()

  // Compartments and Species:
  compartment compartment_;
  species N in compartment_, C in compartment_, B in compartment_;

  // Assignment Rules:
  T := N + C;
  rC := rho_C + b*(T - K_N)^2 / (a*T^2 + (T - K_N)^2);

  // Rate Rules:
  N' = -r_N * N * log((N + C)/K_N);
  C' = -rC * C * log((N + C)/K_C);
  B' = r_B * B - gamma_B * B * (C / (k_B + C));

  // Reactions:

  // Species initializations:
  N = 3.00e9;
  C = 1.80e8;
  B = 9.486e10;

  // Compartment initializations:
  compartment_ = 1;

  // Variable initializations:
  K_N = 2.50e11;
  K_C = 6.96e10;
  r_N = 0.17;
  rho_C = 0.0251;
  a = 42.3;
  b = 52.5;
  r_B = 0.01;
  gamma_B = 1.15;
  k_B = 2.024e9;

  // Other declarations:
  var T, rC;
  const compartment_, K_N, K_C, r_N, rho_C, a, b, r_B, gamma_B, k_B;

  // Unit definitions:
  unit volume = 1e-3 litre;
  unit substance = item;
  unit time_unit = 1 day;

  // Display Names:
  compartment_ is "default compartment";
  N is "Normal T cells";
  C is "CAR T cells";
  B is "Tumour cells";
  T is "Total lymphocyte count";
  rC is "CAR T cell net growth rate function";
  K_N is "Normal T cell carrying capacity";
  K_C is "CAR T cell carrying capacity";
  r_N is "Normal T cell net growth rate";
  rho_C is "Baseline CAR T net growth rate";
  a is "Signalling inefficiency factor in r_C";
  b is "Immune reconstitution impact in r_C";
  r_B is "Tumour net growth rate";
  gamma_B is "Tumour killing rate by CAR T cells";
  k_B is "Killing rate saturation parameter";

end
