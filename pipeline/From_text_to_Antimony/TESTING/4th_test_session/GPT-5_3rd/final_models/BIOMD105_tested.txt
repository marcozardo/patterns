/* Domain: Molecular/cellular systems biology (protein homeostasis, ubiquitin?proteasome system)
   Source: Proctor et al., BMC Systems Biology (2007) ?An in silico model of the ubiquitin-proteasome system that incorporates normal homeostasis and age-related decline?
   Notes:
   - Single-cell, single-compartment model using molecule counts (arbitrary units).
   - ATP/ADP/AMP and ROS are treated as boundary species (kept constant), as described.
   - Polyubiquitin chain lengths up to 8 are modeled; degradation requires chain length >= 4.
   - Deubiquitination proceeds stepwise for both free and proteasome-bound substrates.
   - Aggregation and sequestration pathways included; aggregates can sequester proteasomes.
   - Event included to mimic ts20 E1 shutdown (k62 set to 0 at t = 1800 s).
*/

model *Proctor2007_UPS()

  //// Compartments and Species:
  compartment cell;
  species NatP in cell, MisP in cell, Ub in cell, E1 in cell, E2 in cell, E3 in cell, DUB in cell, Proteasome in cell;
  species $ATP in cell, $ADP in cell, $AMP in cell, $ROS in cell;
  species E3_MisP in cell, E1_Ub in cell, E2_Ub in cell;
  species MisP_Ub in cell, MisP_Ub2 in cell, MisP_Ub3 in cell, MisP_Ub4 in cell, MisP_Ub5 in cell, MisP_Ub6 in cell, MisP_Ub7 in cell, MisP_Ub8 in cell;
  species MisP_Ub4_Proteasome in cell, MisP_Ub5_Proteasome in cell, MisP_Ub6_Proteasome in cell, MisP_Ub7_Proteasome in cell, MisP_Ub8_Proteasome in cell;
  species AggP in cell, SeqAggP in cell, AggP_Proteasome in cell;
  species degUb4 in cell, degUb5 in cell, degUb6 in cell, degUb7 in cell, degUb8 in cell;

  //// Reactions:
  R_synthesis:  => NatP; k1;
  R_misfolding: NatP + ROS => MisP + ROS; k2*NatP*ROS;
  R_refolding:  MisP => NatP; k3*MisP;

  R_E3_bind:    E3 + MisP -> E3_MisP; k61*E3*MisP;
  R_E3_release: E3_MisP -> E3 + MisP; k61r*E3_MisP;

  R_E1_activation: E1 + Ub + ATP => E1_Ub + AMP; k62*E1*Ub*(ATP/(5000 + ATP));
  R_E2_loading:    E2 + E1_Ub    => E2_Ub + E1;  k63*E2*E1_Ub;
  R_monoUb:        E2_Ub + E3_MisP => E2 + E3 + MisP_Ub; k64*E2_Ub*E3_MisP;

  R_poly1: E2_Ub + MisP_Ub  => E2 + MisP_Ub2; k65*E2_Ub*MisP_Ub;
  R_poly2: E2_Ub + MisP_Ub2 => E2 + MisP_Ub3; k65*E2_Ub*MisP_Ub2;

  R_poly4: E2_Ub + MisP_Ub4 => E2 + MisP_Ub5; k65*E2_Ub*MisP_Ub4;
  R_poly5: E2_Ub + MisP_Ub5 => E2 + MisP_Ub6; k65*E2_Ub*MisP_Ub5;
  R_poly6: E2_Ub + MisP_Ub6 => E2 + MisP_Ub7; k65*E2_Ub*MisP_Ub6;
  R_poly7: E2_Ub + MisP_Ub7 => E2 + MisP_Ub8; k65*E2_Ub*MisP_Ub7;

  R_DUB_free8: DUB + MisP_Ub8 => DUB + MisP_Ub7 + Ub; k66*DUB*MisP_Ub8;
  R_DUB_free7: DUB + MisP_Ub7 => DUB + MisP_Ub6 + Ub; k66*DUB*MisP_Ub7;
  R_DUB_free6: DUB + MisP_Ub6 => DUB + MisP_Ub5 + Ub; k66*DUB*MisP_Ub6;
  R_DUB_free5: DUB + MisP_Ub5 => DUB + MisP_Ub4 + Ub; k66*DUB*MisP_Ub5;
  R_DUB_free4: DUB + MisP_Ub4 => DUB + MisP_Ub3 + Ub; k66*DUB*MisP_Ub4;
  R_DUB_free3: DUB + MisP_Ub3 => DUB + MisP_Ub2 + Ub; k66*DUB*MisP_Ub3;
  R_DUB_free2: DUB + MisP_Ub2 => DUB + MisP_Ub  + Ub; k66*DUB*MisP_Ub2;
  R_DUB_free1: DUB + MisP_Ub  => DUB + MisP     + Ub; k66*DUB*MisP_Ub;

  R_bind4: MisP_Ub4 + Proteasome => MisP_Ub4_Proteasome; k67*MisP_Ub4*Proteasome;
  R_bind5: MisP_Ub5 + Proteasome => MisP_Ub5_Proteasome; k67*MisP_Ub5*Proteasome;
  R_bind6: MisP_Ub6 + Proteasome => MisP_Ub6_Proteasome; k67*MisP_Ub6*Proteasome;
  R_bind7: MisP_Ub7 + Proteasome => MisP_Ub7_Proteasome; k67*MisP_Ub7*Proteasome;
  R_bind8: MisP_Ub8 + Proteasome => MisP_Ub8_Proteasome; k67*MisP_Ub8*Proteasome;

  R_DUB_bound8: DUB + MisP_Ub8_Proteasome => DUB + Ub + MisP_Ub7_Proteasome; k68*DUB*MisP_Ub8_Proteasome;
  R_DUB_bound7: DUB + MisP_Ub7_Proteasome => DUB + Ub + MisP_Ub6_Proteasome; k68*DUB*MisP_Ub7_Proteasome;
  R_DUB_bound6: DUB + MisP_Ub6_Proteasome => DUB + Ub + MisP_Ub5_Proteasome; k68*DUB*MisP_Ub6_Proteasome;
  R_DUB_bound5: DUB + MisP_Ub5_Proteasome => DUB + Ub + MisP_Ub4_Proteasome; k68*DUB*MisP_Ub5_Proteasome;
  R_DUB_bound4_release: DUB + MisP_Ub4_Proteasome => DUB + Ub + MisP_Ub3 + Proteasome; k68*DUB*MisP_Ub4_Proteasome;

  R_deg4: ATP + MisP_Ub4_Proteasome => ADP + Proteasome + 4 Ub + degUb4; k69*MisP_Ub4_Proteasome*(ATP/(5000 + ATP));
  R_deg5: ATP + MisP_Ub5_Proteasome => ADP + Proteasome + 5 Ub + degUb5; k69*MisP_Ub5_Proteasome*(ATP/(5000 + ATP));
  R_deg6: ATP + MisP_Ub6_Proteasome => ADP + Proteasome + 6 Ub + degUb6; k69*MisP_Ub6_Proteasome*(ATP/(5000 + ATP));
  R_deg7: ATP + MisP_Ub7_Proteasome => ADP + Proteasome + 7 Ub + degUb7; k69*MisP_Ub7_Proteasome*(ATP/(5000 + ATP));
  R_deg8: ATP + MisP_Ub8_Proteasome => ADP + Proteasome + 8 Ub + degUb8; k69*MisP_Ub8_Proteasome*(ATP/(5000 + ATP));

  R_agg1: 2 MisP => AggP; k71*MisP*MisP;
  R_agg2: MisP + AggP => 2 AggP; k71*MisP*AggP;

  R_agg3: 2 MisP_Ub => AggP; k72*MisP_Ub*MisP_Ub;
  R_agg4: MisP + MisP_Ub => AggP; k72*MisP*MisP_Ub;
  R_agg5: 2 MisP_Ub8 => AggP; k72*MisP_Ub8*MisP_Ub8;
  R_agg6: MisP_Ub7 + MisP_Ub8 => AggP; k72*MisP_Ub7*MisP_Ub8;
  R_agg7: MisP_Ub + AggP => 2 AggP; k72*MisP_Ub*AggP;

  R_sequester: AggP => SeqAggP; k73*AggP;
  R_agg_bind_prot: AggP + Proteasome => AggP_Proteasome; k74*AggP*Proteasome;

  //// Events:
  E1_shutdown: at(time >= 1800): k62 = 0;
