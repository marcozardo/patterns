model *Schmierer2010_HIF_Pathway()

  //// Compartments and Species:
  compartment cell;
  species $H_tot in cell, $H in cell, $H_OH in cell;
  species $A in cell, $A_OH in cell;
  species $P_tot in cell, $F_tot in cell, $O2 in cell;

  //// Assignment Rules:
  H_OH := H_tot - H;
  A_tot := A + A_OH;
  v_P := k_cat_prime_P * P_tot * (1/(K_D_hat_P + H_p + P_tot)) * (O2_tilde/(1+O2_tilde));
  H_p := -0.5 * (A_tot - P_tot - K_D_hat_P + sqrt((K_D_hat_P + P_tot - A_tot)^2 + 4*K_D_hat_P*A_tot));
  v_FH := k_cat_prime_FH * F_tot * (1/(K_i_FH + H_F + F_tot)) * (O2_tilde/(alpha+O2_tilde));
  H_F := 0.5 * (A - F_tot - K_i_FH + sqrt((K_i_FH + F_tot - A)^2 + 4*K_i_FH*A));
  K_i_FH := K_D_hat_FH * (1 + A/K_D_hat_FA + gamma*A_OH/K_D_hat_FA);
  v_FA := (k_cat_prime_FA * F_tot) / (K_D_hat_FA*(1 + H_F/K_D_hat_FH) + A + gamma*A_OH) * (O2_tilde/(alpha+O2_tilde));
  F_free := 1/(1 + kappa*A);
  kappa := A_tot/K_D_hat_FA;
  beta := (k_cat_prime_FA * F_tot)/(k_d_A * A_tot);

  //// Rate Rules:
  H_tot' = 1 - H_tot*(1 + v_P);
  H' = 1 - H*(1 + v_P + v_FH);
  A' = (A_tot - A*(1 + epsilon*v_FA))/epsilon;

  //// Reactions:
  // Note: Hydroxylation reactions are implemented via rate rules above

  //// Species initializations:
  H_tot = 0;
  H = 0;
  A = 0;
  A_OH = 0;

  //// Compartment initializations:
  cell = 1;

  //// Variable initializations:
  // Dimensionless parameters from Table 2
  P_tot = 0.2;
  K_D_hat_P = 1;
  k_cat_prime_P = 500;
  F_tot = 1;
  K_D_hat_FH = 1;
  alpha = 0.33;
  A_tot = 100;
  K_D_hat_FA = 1;
  gamma = 0.02;
  epsilon = 5;
  O2_tilde = 1;
  k_cat_prime_FH = 500;
  k_cat_prime_FA = 500;
  k_d_A = 1;

  //// Other declarations:
  var H_tot, H, H_OH, A, A_OH, v_P, v_FH, v_FA, H_p, H_F, K_i_FH, F_free, kappa, beta;
  const cell, P_tot, F_tot, O2_tilde, K_D_hat_P, k_cat_prime_P, K_D_hat_FH, alpha, A_tot, K_D_hat_FA, gamma, epsilon, k_cat_prime_FH, k_cat_prime_FA, k_d_A;

  //// Unit definitions:
  unit substance = 1e-9 mole;
  unit time_unit = 1 second;

  //// Display Names:
  cell is "Intracellular compartment";
  H_tot is "Total HIF?";
  H is "Non-CAD-hydroxylated HIF?";
  H_OH is "CAD-hydroxylated HIF?";
  A is "Unhydroxylated ARD proteins";
  A_OH is "Hydroxylated ARD proteins";
  P_tot is "Total PHD enzyme";
  F_tot is "Total FIH enzyme";
  O2_tilde is "Oxygen concentration (relative to PHD K_M)";
  v_P is "PHD-dependent HIF? ODD-hydroxylation rate";
  v_FH is "FIH-dependent HIF? CAD-hydroxylation rate";
  v_FA is "FIH-dependent ARD protein hydroxylation rate";
end
